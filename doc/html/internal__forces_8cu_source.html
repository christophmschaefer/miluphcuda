400 
401  // new implementation (after july 2017)
402  for (e = 0; e < DIM; e++) {
403  for (f = 0; f < DIM; f++) {
404  for (kk = 0; kk < DIM; kk++) {

409  (-dv[f]) * dr[kk] * dWdr/r);
410 
411  rdot[e][f] += 0.5 * p.m[j]/p.rho[j] *
412  (p_rhs.tensorialCorrectionMatrix[i*DIM*DIM+f*DIM+kk] *
413  (-dv[e]) * dr[kk] * dWdr/r
414  - p_rhs.tensorialCorrectionMatrix[i*DIM*DIM+e*DIM+kk] *
415  (-dv[f]) * dr[kk] * dWdr/r);
416  }
417  }
418  }
419 # else

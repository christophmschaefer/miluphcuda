57 __global__ void CorrectorStep()
58 {
59  register int i;
60 #if SOLID
61  register int j;
62  register int k;
63 #endif
64 
65 #if GRAVITATING_POINT_MASSES
66  // pointmass loop
67  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numPointmasses; i+= blockDim.x * gridDim.x) {
68  pointmass.x[i] = pointmass.x[i] + dt * predictor_pointmass.vx[i];
69 #if DIM > 1
70  pointmass.y[i] = pointmass.y[i] + dt * predictor_pointmass.vy[i];
71  pointmass.vy[i] = pointmass.vy[i] + dt * predictor_pointmass.ay[i];
72  pointmass.ay[i] = predictor_pointmass.ay[i];
73 #endif
74  pointmass.vx[i] = pointmass.vx[i] + dt * predictor_pointmass.ax[i];
75  pointmass.ax[i] = predictor_pointmass.ax[i];
76 #if DIM == 3
77  pointmass.z[i] = pointmass.z[i] + dt * predictor_pointmass.vz[i];
78  pointmass.vz[i] = pointmass.vz[i] + dt * predictor_pointmass.az[i];
79  pointmass.az[i] = predictor_pointmass.az[i];
80 #endif
81  }
82 #endif // GRAVITATING_POINT_MASSES
83 
84  // particle loop
85  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i+= blockDim.x * gridDim.x) {
86  p.x[i] = p.x[i] + dt * predictor.dxdt[i];
87 #if DIM > 1
88  p.y[i] = p.y[i] + dt * predictor.dydt[i];
89  p.vy[i] = p.vy[i] + dt * predictor.ay[i];
90  p.ay[i] = predictor.ay[i];
91 #endif
92  p.vx[i] = p.vx[i] + dt * predictor.ax[i];
93  p.ax[i] = predictor.ax[i];
94 #if DIM == 3
95  p.z[i] = p.z[i] + dt * predictor.dzdt[i];
96  p.vz[i] = p.vz[i] + dt * predictor.az[i];
97  p.az[i] = predictor.az[i];
98 #endif
99 #if INTEGRATE_DENSITY
100  p.rho[i] = p.rho[i] + dt * predictor.drhodt[i];
101  p.drhodt[i] = predictor.drhodt[i];
102 #else
103  p.rho[i] = p.rho[i];
104 #endif
105 #if INTEGRATE_ENERGY
106  p.e[i] = p.e[i] + dt * predictor.dedt[i];
107  p.dedt[i] = predictor.dedt[i];
108 #endif
109 #if FRAGMENTATION
110  p.d[i] = p.d[i] + dt * predictor.dddt[i];
111  p.dddt[i] = predictor.dddt[i];
112 #endif
113 #if INTEGRATE_SML
114  p.h[i] = p.h[i] + dt * predictor.dhdt[i];
115  p.dhdt[i] = predictor.dhdt[i];
116 #else
117  p.h[i] = predictor.h[i];
118 #endif
119 #if JC_PLASTICITY
120  p.ep[i] = p.ep[i] + dt * predictor.edotp[i];
121  p.edotp[i] = predictor.edotp[i];
122  p.T[i] = p.T[i] + dt * predictor.dTdt[i];
123  p.dTdt[i] = predictor.dTdt[i];
124 #endif
125 #if SIRONO_POROSITY
126  p.rho_0prime[i] = p.rho_0prime[i];
127  p.rho_c_plus[i] = p.rho_c_plus[i];
128  p.rho_c_minus[i] = p.rho_c_minus[i];
129  p.compressive_strength[i] = p.compressive_strength[i];
130  p.tensile_strength[i] = p.tensile_strength[i];
131  p.shear_strength[i] = p.shear_strength[i];
132  p.K[i] = p.K[i];
133  p.flag_rho_0prime[i] = p.flag_rho_0prime[i];
134  p.flag_plastic[i] = p.flag_plastic[i];
135 #endif
136 #if EPSALPHA_POROSITY
137  p.alpha_epspor[i] = p.alpha_epspor[i] + dt * predictor.dalpha_epspordt[i];
138  p.epsilon_v[i] = p.epsilon_v[i] + dt * predictor.depsilon_vdt[i];
139  p.dalpha_epspordt[i] = predictor.dalpha_epspordt[i];
140  p.depsilon_vdt[i] = predictor.depsilon_vdt[i];
141 #endif
142 #if INVISCID_SPH
143  p.beta[i] = p.beta[i] + dt * predictor.dbetadt[i];
144  p.dbetadt[i] = predictor.dbetadt[i];
145 #endif
146 #if SOLID
147  for (j = 0; j < DIM; j++) {
148  for (k = 0; k < DIM; k++) {
149  p.S[stressIndex(i,j,k)] = p.S[stressIndex(i,j,k)] + dt * predictor.dSdt[stressIndex(i,j,k)];
150  p.dSdt[stressIndex(i,j,k)] = predictor.dSdt[stressIndex(i,j,k)];
151  }
152  }
153 #endif
154  }

157 __global__ void PredictorStep()
158 {
159  register int i;
160 #if SOLID
161  register int j;
162  register int k;
163 #endif
164 
165 
166 #if GRAVITATING_POINT_MASSES
167  // pointmass loop
168  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numPointmasses; i+= blockDim.x * gridDim.x) {
169  predictor_pointmass.x[i] = pointmass.x[i] + dt/2 * pointmass.vx[i];
170  predictor_pointmass.vx[i] = pointmass.vx[i] + dt/2 * pointmass.ax[i];
171 
172 #if DIM > 1
173  predictor_pointmass.y[i] = pointmass.y[i] + dt/2 * pointmass.vy[i];
174  predictor_pointmass.vy[i] = pointmass.vy[i] + dt/2 * pointmass.ay[i];
175 #endif
176 #if DIM > 2
177  predictor_pointmass.z[i] = pointmass.z[i] + dt/2 * pointmass.vz[i];
178  predictor_pointmass.vz[i] = pointmass.vz[i] + dt/2 * pointmass.az[i];
179 #endif
180  }
181 #endif // GRAVITATING_POINT_MASSES
182 
183 
184  // particle loop
185  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i+= blockDim.x * gridDim.x) {
186  predictor.x[i] = p.x[i] + dt/2 * p.dxdt[i];
187  predictor.vx[i] = p.vx[i] + dt/2 * p.ax[i];
188 
189 #if DIM > 1
190  predictor.y[i] = p.y[i] + dt/2 * p.dydt[i];
191  predictor.vy[i] = p.vy[i] + dt/2 * p.ay[i];
192 #endif
193 #if DIM > 2
194  predictor.z[i] = p.z[i] + dt/2 * p.dzdt[i];
195  predictor.vz[i] = p.vz[i] + dt/2 * p.az[i];
196 #endif
197 #if INTEGRATE_DENSITY
198  predictor.rho[i] = p.rho[i] + dt/2 * p.drhodt[i];
199 #else
200  predictor.rho[i] = p.rho[i];
201 #endif
202 #if INTEGRATE_ENERGY
203  predictor.e[i] = p.e[i] + dt/2 * p.dedt[i];
204 #endif
205 #if INTEGRATE_SML
206  predictor.h[i] = p.h[i] + dt/2 * p.dhdt[i];
207 #else
208  predictor.h[i] = p.h[i];
209 #endif
210 
211 #if PALPHA_POROSITY
212  // p points to p_device and p.p is the pressure at the start of the timestep
213  // while predictor changes during the adaptive time step
214  predictor.pold[i] = p.p[i];
215  predictor.alpha_jutzi[i] = p.alpha_jutzi[i] + dt/2 * p.dalphadt[i];
216  predictor.alpha_jutzi_old[i] = p.alpha_jutzi_old[i];
217 #endif
218 
219 #if FRAGMENTATION
220  predictor.d[i] = p.d[i] + dt/2 * p.dddt[i];
221  predictor.numActiveFlaws[i] = p.numActiveFlaws[i];
222 #if PALPHA_POROSITY
223  predictor.damage_porjutzi[i] = p.damage_porjutzi[i] + dt/2 * p.ddamage_porjutzidt[i];
224 #endif
225 #endif
226 #if JC_PLASTICITY
227  predictor.ep[i] = p.ep[i] + dt/2 * p.edotp[i];
228  predictor.T[i] = p.T[i] + dt/2 * p.dTdt[i];
229 #endif
230 #if SIRONO_POROSITY
231  predictor.rho_0prime[i] = p.rho_0prime[i];
232  predictor.rho_c_plus[i] = p.rho_c_plus[i];
233  predictor.rho_c_minus[i] = p.rho_c_minus[i];
234  predictor.compressive_strength[i] = p.compressive_strength[i];
235  predictor.tensile_strength[i] = p.tensile_strength[i];
236  predictor.shear_strength[i] = p.shear_strength[i];
237  predictor.K[i] = p.K[i];
238  predictor.flag_rho_0prime[i] = p.flag_rho_0prime[i];
239  predictor.flag_plastic[i] = p.flag_plastic[i];
240 #endif
241 #if EPSALPHA_POROSITY
242  predictor.alpha_epspor[i] = p.alpha_epspor[i] + dt/2 * p.dalpha_epspordt[i];
243  predictor.epsilon_v[i] = p.epsilon_v[i] + dt/2 * p.depsilon_vdt[i];
244 #endif
245 #if INVISCID_SPH
246  predictor.beta[i] = p.beta[i] + dt/2 * p.dbetadt[i];
247 #endif
248 #if SOLID
249  for (j = 0; j < DIM; j++) {
250  for (k = 0; k < DIM; k++) {
251  predictor.S[stressIndex(i,j,k)] = p.S[stressIndex(i,j,k)] + dt/2 * p.dSdt[stressIndex(i,j,k)];
252  }
253  }
254 #endif
255  }

344 /* check the pressure change to avoid large deviation from the crush-curve */
345 __global__ void pressureChangeCheck(double *maxpressureDiffPerBlock)
346 {
347  __shared__ double sharedMaxpressureDiff[NUM_THREADS_PC_INTEGRATOR];
348  double localMaxpressureDiff = 0.0;
349  double tmp = 0.0;
350  int i, j, k, m;
351  maxpressureDiff = 0.0;
352  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i+= blockDim.x * gridDim.x) {
353  if (matEOS[p_rhs.materialId[i]] == EOS_TYPE_JUTZI || matEOS[p_rhs.materialId[i]] == EOS_TYPE_JUTZI_MURNAGHAN) {
354  // cms - 20190626
355  // first rhs is called at beginning of timestep with predictor
356  // and at the end with p_device
357  tmp = (p.p[i] - predictor.pold[i]);
358  localMaxpressureDiff = max(tmp, localMaxpressureDiff);
359  }
360  }
361  i = threadIdx.x;
362  sharedMaxpressureDiff[i] = localMaxpressureDiff;
363  for (j = NUM_THREADS_PC_INTEGRATOR / 2; j > 0; j /= 2) {
364  __syncthreads();
365  if (i < j) {
366  k = i + j;
367  sharedMaxpressureDiff[i] = localMaxpressureDiff = max(localMaxpressureDiff, sharedMaxpressureDiff[k]);
368  }
369  }
370  // write block result to global memory
371  if (i == 0) {
372  k = blockIdx.x;
373  maxpressureDiffPerBlock[k] = localMaxpressureDiff;
374  m = gridDim.x - 1;
375  if (m == atomicInc((unsigned int *)&blockCount, m)) {
376  // last block, so combine all block results
377  for (j = 0; j <= m; j++) {
378  localMaxpressureDiff = max(localMaxpressureDiff, maxpressureDiffPerBlock[j]);
379  }
380  maxpressureDiff = localMaxpressureDiff;
381  // reset block count
382  blockCount = 0;
383  }
384  if (maxpressureDiff > max_abs_pressure_change) {
385  printf("maxpressure change %e\n", maxpressureDiff);
386  pressureChangeSmallEnough = FALSE;
387  dt = 0.25 * dt;
388  dt = min(dt, endTimeD - currentTimeD);
389  } else {
390  pressureChangeSmallEnough = TRUE;
391  // currentTimeD += dt;
392  }
393  }
394 }

397 __global__ void setTimestep(double *forcesPerBlock, double *courantPerBlock, double *dtSPerBlock, double *dtePerBlock, double *dtrhoPerBlock, double *dtdamagePerBlock, double *dtalphaPerBlock, double *dtartviscPerBlock, double *dtbetaPerBlock, double *dtalpha_epsporPerBlock, double *dtepsilon_vPerBlock)
398 {
399 
400 #define SAFETY_FIRST 0.7
401 
402  __shared__ double sharedForces[NUM_THREADS_LIMITTIMESTEP];
403  __shared__ double sharedCourant[NUM_THREADS_LIMITTIMESTEP];
404  __shared__ double sharedArtVisc[NUM_THREADS_LIMITTIMESTEP];
405  __shared__ double sharedS[NUM_THREADS_LIMITTIMESTEP];
406  __shared__ double sharede[NUM_THREADS_LIMITTIMESTEP];
407  __shared__ double sharedrho[NUM_THREADS_LIMITTIMESTEP];
408  __shared__ double shareddamage[NUM_THREADS_LIMITTIMESTEP];
409  __shared__ double sharedalpha[NUM_THREADS_LIMITTIMESTEP];
410  __shared__ double sharedbeta[NUM_THREADS_LIMITTIMESTEP];
411  __shared__ double sharedalpha_epspor[NUM_THREADS_LIMITTIMESTEP];
412  __shared__ double sharedepsilon_v[NUM_THREADS_LIMITTIMESTEP];
413 
414  int i, j, k, m;
415  int d, dd;
416  int index;
417  int hasEnergy;
418  double forces = DBL_MAX, courant = DBL_MAX;
419  double dtx = DBL_MAX;
420  double dtS = DBL_MAX;
421  double dtrho = DBL_MAX;
422  double dte = DBL_MAX;
423  double dtdamage = DBL_MAX;
424  double dtalpha = DBL_MAX;
425  double dtbeta = DBL_MAX;
426  double dtalpha_epspor = DBL_MAX;
427  double dtepsilon_v = DBL_MAX;
428  double temp;
429  double sml;
430  int matId;
431 #if SOLID
432  double myS, dS;
433 #endif
434  double ax, ay;
435 #if DIM == 3
436  double az;
437 #endif
438  double dtartvisc = DBL_MAX;
439 
440  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i+= blockDim.x * gridDim.x) {
441  matId = p_rhs.materialId[i];
442 #if INTEGRATE_ENERGY
443  hasEnergy = 0;
444 
445  switch (matEOS[matId]) {
446  case (EOS_TYPE_TILLOTSON):
447  hasEnergy = 1;
448  break;
449  case (EOS_TYPE_JUTZI):
450  hasEnergy = 1;
451  break;
452  case (EOS_TYPE_SIRONO):
453  hasEnergy = 1;
454  break;
455  case (EOS_TYPE_EPSILON):
456  hasEnergy = 1;
457  break;
458  case (EOS_TYPE_ANEOS):
459  hasEnergy = 1;
460  break;
461  default:
462  hasEnergy = 0;
463  break;
464  }
465 #endif
466  ax = p.ax[i];
467 #if DIM > 1
468  ay = p.ay[i];
469 #endif
470 #if DIM == 3
471  az = p.az[i];
472 #endif
473  temp = ax*ax;
474 #if DIM > 1
475  temp += + ay*ay;
476 #endif
477 #if DIM == 3
478  temp += az*az;
479 #endif
480  ;
481  sml = p.h[i];
482  temp = sqrt(sml / sqrt(temp));
483  forces = min(forces, temp);
484  temp = sml / p.cs[i];
485  courant = min(courant, temp);
486 
487 #define CFL 0.7
488 
489 #if ARTIFICIAL_VISCOSITY
490  temp = CFL * sml / (p.cs[i] + 1.2 * (matAlpha[matId]) * p.cs[i] + matBeta[matId] *
491  p.muijmax[i]);
492  dtartvisc = min(dtartvisc, temp);
493 #endif
494 #if INVISCID_SPH
495  if (p.dbetadt[i] != 0) {
496  temp = SAFETY_FIRST * (fabs(p.beta[i])+betamin_d)/fabs(p.dbetadt[i]);
497  dtbeta = min(temp, dtbeta);
498  }
499 #endif
500 #if SOLID
501  myS = 0;
502  dS = 0;
503 
504  for (d = 0; d < DIM; d++) {
505  for (dd = 0; dd < DIM; dd++) {
506  index = i*DIM*DIM+d*DIM+dd;
507  myS += p.S[index]*p.S[index];
508  dS += p.dSdt[index]*p.dSdt[index];
509  }
510  }
511  if (dS != 0) {
512  temp = SAFETY_FIRST * sqrt((myS+Smin_d)/dS);
513  dtS = min(temp, dtS);
514  }
515 #endif
516 #if INTEGRATE_DENSITY
517  if (p.drhodt[i] != 0) {
518  temp = SAFETY_FIRST * (fabs(p.rho[i])+rhomin_d)/fabs(p.drhodt[i]);
519  dtrho = min(temp, dtrho);
520  }
521 #endif
522 #if INTEGRATE_ENERGY
523  if (p.dedt[i] != 0 && hasEnergy) {
524  temp = SAFETY_FIRST * (fabs(p.e[i])+emin_d)/fabs(p.dedt[i]);
525  dte = min(temp, dte);
526  }
527 #endif
528 
529 #if PALPHA_POROSITY
530  if (p.dalphadt[i] != 0) {
531  temp = 1.0e-2 / fabs(p.dalphadt[i]);
532  dtalpha = min(temp, dtalpha);
533  }
534 #endif
535 
536 #if EPSALPHA_POROSITY
537  if (p.dalpha_epspordt[i] != 0) {
538  temp = 1.0e-1 / fabs(p.dalpha_epspordt[i]);
539  dtalpha_epspor = min(temp, dtalpha_epspor);
540  }
541 
542  if (p.depsilon_vdt[i] != 0) {
543  temp = SAFETY_FIRST * (fabs(p.epsilon_v[i])+epsilon_vmin_d)/fabs(p.depsilon_vdt[i]);
544  dtepsilon_v = min(temp, dtepsilon_v);
545  }
546 #endif
547 
548 #if FRAGMENTATION
549  if (p.dddt[i] != 0) {
550  temp = SAFETY_FIRST * (fabs(p.d[i])+damagemin_d)/fabs(p.dddt[i]);
551  dtdamage = min(temp, dtdamage);
552  }
553 #endif
554 
555 
556  }
557  i = threadIdx.x;
558  sharedForces[i] = forces;
559  sharedCourant[i] = courant;
560  sharedS[i] = dtS;
561  sharede[i] = dte;
562  sharedrho[i] = dtrho;
563  shareddamage[i] = dtdamage;
564  sharedalpha[i] = dtalpha;
565  sharedbeta[i] = dtbeta;
566  sharedalpha_epspor[i] = dtalpha_epspor;
567  sharedepsilon_v[i] = dtepsilon_v;
568 
569 #if ARTIFICIAL_VISCOSITY
570  sharedArtVisc[i] = dtartvisc;
571 #endif
572  for (j = NUM_THREADS_LIMITTIMESTEP / 2; j > 0; j /= 2) {
573  __syncthreads();
574  if (i < j) {
575  k = i + j;
576  sharedForces[i] = forces = min(forces, sharedForces[k]);
577  sharedCourant[i] = courant = min(courant, sharedCourant[k]);
578  sharedS[i] = dtS = min(dtS, sharedS[k]);
579  sharede[i] = dte = min(dte, sharede[k]);
580  sharedrho[i] = dtrho = min(dtrho, sharedrho[k]);
581  shareddamage[i] = dtdamage = min(dtdamage, shareddamage[k]);
582  sharedalpha[i] = dtalpha = min(dtalpha, sharedalpha[k]);
583  sharedalpha_epspor[i] = dtalpha_epspor = min(dtalpha_epspor, sharedalpha_epspor[k]);
584  sharedepsilon_v[i] = dtepsilon_v = min(dtepsilon_v, sharedepsilon_v[k]);
585 #if ARTIFICIAL_VISCOSITY
586  sharedArtVisc[i] = dtartvisc = min(dtartvisc, sharedArtVisc[k]);
587 #endif
588 #if INVISCID_SPH
589  sharedbeta[i] = dtbeta = min(dtbeta, sharedbeta[i]);
590 #endif
591  }
592  }
593  // write block result to global memory
594  if (i == 0) {
595  k = blockIdx.x;
596  forcesPerBlock[k] = forces;
597  courantPerBlock[k] = courant;
598  dtSPerBlock[k] = dtS;
599  dtePerBlock[k] = dte;
600  dtrhoPerBlock[k] = dtrho;
601  dtdamagePerBlock[k] = dtdamage;
602  dtalphaPerBlock[k] = dtalpha;
603  dtalpha_epsporPerBlock[k] = dtalpha_epspor;
604  dtepsilon_vPerBlock[k] = dtepsilon_v;
605 #if ARTIFICIAL_VISCOSITY
606  dtartviscPerBlock[k] = dtartvisc;
607 #endif
608 #if INVISCID_SPH
609  dtbetaPerBlock[k] = dtbeta;
610 #endif
611  m = gridDim.x - 1;
612  if (m == atomicInc((unsigned int *)&blockCount, m)) {
613  // last block, so combine all block results
614  for (j = 0; j <= m; j++) {
615  forces = min(forces, forcesPerBlock[j]);
616  courant = min(courant, courantPerBlock[j]);
617  dtS = min(dtS, dtSPerBlock[j]);
618  dte = min(dte, dtePerBlock[j]);
619  dtrho = min(dtrho, dtrhoPerBlock[j]);
620  dtdamage = min(dtdamage, dtdamagePerBlock[j]);
621  dtalpha = min(dtalpha, dtalphaPerBlock[j]);
622  dtalpha_epspor = min(dtalpha_epspor, dtalpha_epsporPerBlock[j]);
623  dtepsilon_v = min(dtepsilon_v, dtepsilon_vPerBlock[j]);
624 #if ARTIFICIAL_VISCOSITY
625  dtartvisc = min(dtartvisc, dtartviscPerBlock[j]);
626 #endif
627 #if INVISCID_SPH
628  dtbeta = min(dtbeta, dtbetaPerBlock[j]);
629 #endif
630  }
631  // set new timestep
632  dt = dtx = min(COURANT*courant, forces*0.2);
633 #if SOLID
634  dt = min(dt, dtS);
635 #endif
636 #if INTEGRATE_ENERGY
637  dt = min(dt, dte);
638 #endif
639 #if INTEGRATE_DENSITY
640  dt = min(dt, dtrho);
641 #endif
642 #if FRAGMENTATION
643  dt = min(dt, dtdamage);
644 #endif
645 #if PALPHA_POROSITY
646  dt = min(dt, dtalpha);
647 #endif
648 #if EPSALPHA_POROSITY
649  dt = min(dt, dtalpha_epspor);
650  dt = min(dt, dtepsilon_v);
651 #endif
652 #if ARTIFICIAL_VISCOSITY
653  dt = min(dt, dtartvisc);
654 #endif
655 #if INVISCID_SPH
656  dt = min(dt, dtbeta);
657 #endif
658  dt = min(dt, endTimeD - currentTimeD);
659  printf("Time Step Information: dt(v and x): %e dtS: %e dte: %e dtrho: %e dtdamage: %e dtalpha: %e dtalpha_epspor: %e dtepsilon_v: %e\n",
660  dtx, dtS, dte, dtrho, dtdamage, dtalpha, dtalpha_epspor, dtepsilon_v);
661  printf("time: %e timestep set to %e, integrating until %e \n", currentTimeD, dt, endTimeD);
662 #if !PALPHA_POROSITY
663  currentTimeD += dt;
664 #endif
665  // reset block count
666  blockCount = 0;
667  }
668  }

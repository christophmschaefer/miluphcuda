29 // transform to principal axes before applying artificial stress
30 #define PRINCIPAL_AXES_ARTIFICIAL_STRESS 1
31 
32 #if ARTIFICIAL_STRESS
33 __global__ void compute_artificial_stress(int *interactions)
34 {
35  int i, inc;
36  int d, e;
37  int niters = 0;
38  int matId;
39  double max_ev = -1e300;
40 
41  // the diagonalized tensors
42  double main_stresses[DIM];
43  double rotation_matrix[DIM][DIM];
44  double R[DIM][DIM];
45  double sigma[DIM][DIM];
46  double Rtmp[DIM][DIM];
47 
48 
49  inc = blockDim.x * gridDim.x;
50 
51  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
52  matId = p_rhs.materialId[i];
53  // build stress tensor from deviatoric stress and pressure
54  for (d = 0; d < DIM; d++) {
55  for (e = 0; e < DIM; e++) {
56  sigma[d][e] = p_rhs.sigma[stressIndex(i,d,e)];
57  }
58  }
59 #if PRINCIPAL_AXES_ARTIFICIAL_STRESS
60  niters = calculate_all_eigenvalues(sigma, main_stresses, rotation_matrix);
61  // determine the maximum stress
62  max_ev = main_stresses[0];
63  for (e = 1; e < DIM; e++) {
64  if (main_stresses[e] > max_ev) {
65  max_ev = main_stresses[e];
66  }
67  }
68  // now calculate the artificial stress from the main stresses
69  for (d = 0; d < DIM; d++) {
70  for (e = 0; e < DIM; e++) {
71  R[d][e] = 0.0;
72  }
73  if (main_stresses[d] > 0) {
74  R[d][d] = -matepsilon_stress[matId]*main_stresses[d];
75  }
76  }
77  // convert back in the original coordinate system with the rotation matrix
78  multiply_matrix(R, rotation_matrix, Rtmp);
79  transpose_matrix(rotation_matrix);
80  multiply_matrix(rotation_matrix, Rtmp, R);
81  // now save R for the particle
82  for (d = 0; d < DIM; d++) {
83  for (e = 0; e < DIM; e++) {
84  p_rhs.R[stressIndex(i,d,e)]= R[d][e];
85  }
86  }
87 #else
88  // no transformation, just reduce the standard stress
89  for (d = 0; d < DIM; d++) {
90  for (e = 0; e < DIM; e++) {
91  if (sigma[d][e] > 0) {
92  p_rhs.R[stressIndex(i,d,e)] = -matepsilon_stress[matId] * sigma[d][e];
93  }
94  }
95  }
96 #endif // PRINCIPAL_AXES_ARTIFICIAL_STRESS
97 
98  }
99 }
100 #endif

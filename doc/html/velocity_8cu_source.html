30 __global__ void setlocationchanges(int *interactions)
31 {
32 
33  register int i, inc;
34  inc = blockDim.x * gridDim.x;
35  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
36  if (EOS_TYPE_IGNORE == matEOS[p_rhs.materialId[i]] || p_rhs.materialId[i] == EOS_TYPE_IGNORE) {
37  continue;
38  }
39  p.dxdt[i] = p.vx[i];
40 #if DIM > 1
41  p.dydt[i] = p.vy[i];
42 #endif
43 #if XSPH
44  p.dxdt[i] += 0.5 * p.xsphvx[i];
45 #if DIM > 1
46  p.dydt[i] += 0.5 * p.xsphvy[i];
47 #endif
48 #endif
49 #if DIM == 3
50  p.dzdt[i] = p.vz[i];
51 #if XSPH
52  p.dzdt[i] += 0.5 * p.xsphvz[i];
53 #endif
54 #endif
55  }
56 }

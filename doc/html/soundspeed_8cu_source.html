29 __global__ void calculateSoundSpeed()
30 {
31  register int i, inc, matId;
32  int d;
33  int j;
34  double m_com;
35 
36  inc = blockDim.x * gridDim.x;
37  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
38  matId = p_rhs.materialId[i];
39  if (EOS_TYPE_POLYTROPIC_GAS == matEOS[matId]) {
40  p.cs[i] = sqrt(matPolytropicK[matId] * pow(p.rho[i], matPolytropicGamma[matId]-1.0));
41  } else if (EOS_TYPE_LOCALLY_ISOTHERMAL_GAS == matEOS[matId]) {
42  // p = vkep \times scale_height
43  double distance = 0.0;
44  distance = p.x[i] * p.x[i];
45 #if DIM > 1
46  distance += p.y[i]*p.y[i];
47 #if DIM > 2
48  distance += p.z[i]*p.z[i];
49 #endif
50 #endif
51  distance = sqrt(distance);
52  m_com = 0;
53  for (j = 0; j < numPointmasses; j++) {
54  m_com += pointmass.m[j];
55  }
56 
57  double vkep = sqrt(C_GRAVITY_SI * m_com/distance);
58  p.cs[i] = vkep * scale_height;
59  } else if (EOS_TYPE_IDEAL_GAS == matEOS[matId]) {
60  p.cs[i] = sqrt(matPolytropicGamma[matId] * p.p[i] / p.rho[i]);
61  } else if (EOS_TYPE_JUTZI == matEOS[matId] || EOS_TYPE_JUTZI_MURNAGHAN == matEOS[matId]) {
62 #if PALPHA_POROSITY
63  //p.cs[i] = sqrt(matBulkmodulus[matId]/matTillRho0[matId]);
64 // if (p.alpha_jutzi[i] > 1.0 && abs(p.dalphadp[i]) > 0) {

66 // p.cs[i] = sqrt((p.alpha_jutzi[i] * p.delpdelrho[i] + p.delpdele[i] * p.p[i] / (p.rho[i] * p.rho[i]))
67 // / (p.alpha_jutzi[i] + p.dalphadp[i] * (p.p[i] - p.rho[i] * p.delpdelrho[i])));
68 // }
69 // }
70 // if (!isnan(p.cs[i])) {
71 // p_rhs.cs_old[i] = p.cs[i];
72 // } else {
73 // p.cs[i] = p_rhs.cs_old[i];
74 // }
75 //#if 0
76  /* switched from jutzis implementation of the soundspeed to a linear soundspeed from cs_porous with alpha=alpha0 to cs_solid with alpha=1 (also done in iSale) */
77  p.cs[i] = matcs_solid[matId] + (matcs_porous[matId] - matcs_solid[matId]) * (p.alpha_jutzi[i] - 1.0) / (matporjutzi_alpha_0[matId] - 1.0);
78 #if DEBUG
79  if (isnan(p.cs[i])) {
80  printf("i %d alpha_jutzi %e delpdelrho %e delpdele %e dalphadp %e p %e rho %e\n", i, p.alpha_jutzi[i], p.delpdelrho[i], p.delpdele[i], p.dalphadp[i], p.p[i], p.rho[i]);
81  assert(0);
82  }
83 //#endif
84 #endif
85 
86 #endif
87 #if SIRONO_POROSITY
88  } else if (EOS_TYPE_SIRONO == matEOS[matId]) {
89  if (p.flag_plastic[i] > 0)
90  p.cs[i] = sqrt(p.compressive_strength[i] / p.rho[i]);
91  else
92  p.cs[i] = sqrt(p.K[i] / p.rho_0prime[i]);
93 #endif
94 #if EPSALPHA_POROSITY
95  /* Improvements to epsilon-alpha model by Collins et al 2010 */
96  } else if (EOS_TYPE_EPSILON == matEOS[matId]) {
97  double c_s0 = sqrt(matBulkmodulus[matId]/matTillRho0[matId]);
98  double c_p0 = sqrt(matBulkmodulus[matId]/(matTillRho0[matId] / matporepsilon_alpha_0[matId]));
99  p.cs[i] = c_s0 + (p.alpha_epspor[i] - 1.0) / (matporepsilon_alpha_0[matId] - 1.0) * (c_p0 - c_s0);
100 #endif
101  }
102  // other material types have a constant soundspeed which is calculated in initializeSoundspeed
103  }

108 __global__ void initializeSoundspeed()
109 {
110  register int i, inc, matId;
111  inc = blockDim.x * gridDim.x;
112  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
113  matId = p_rhs.materialId[i];
114  if (EOS_TYPE_POLYTROPIC_GAS == matEOS[matId]) {
115  p.cs[i] = 0.0; // for gas this will be calculated each step by kernel calculateSoundSpeed
116  } else if (EOS_TYPE_ISOTHERMAL_GAS == matEOS[matId]) {
117  /* this is pure molecular hydrogen at 10 K */
118  p.cs[i] = 203;
119  } else if (EOS_TYPE_TILLOTSON == matEOS[matId]) {
120  p.cs[i] = sqrt(matBulkmodulus[matId]/matTillRho0[matId]);
121  } else if (EOS_TYPE_ANEOS == matEOS[matId]) {
122  p.cs[i] = aneos_bulk_cs_c[matId];
123  } else if (EOS_TYPE_MURNAGHAN == matEOS[matId]) {
124  p.cs[i] = sqrt(matBulkmodulus[matId]/matRho0[matId]);
125  } else if (EOS_TYPE_REGOLITH == matEOS[matId]) {
126  //sound speed in soil is typically between 450 and 600 m/s according to Ha H. Bui 2008
127  p.cs[i] = 500.0;
128 // } else if (EOS_TYPE_EPSILON == matEOS[matId]) {
129 // p.cs[i] = sqrt(matBulkmodulus[matId]/matTillRho0[matId]);
130  }
131  }
132 }

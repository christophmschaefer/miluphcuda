299  __shared__ double sharedMaxpressureDiff[NUM_THREADS_PC_INTEGRATOR];

305  if (matEOS[p_rhs.materialId[i]] == EOS_TYPE_JUTZI || matEOS[p_rhs.materialId[i]] == EOS_TYPE_JUTZI_MURNAGHAN || matEOS[p_rhs.materialId[i]] == EOS_TYPE_JUTZI_ANEOS) {
306  /* when this function is called, p points to predictor_device */
307  // only consider pressure if the particle is compressed

309  tmp = (p.p[i] - p.pold[i]);
310  } else {
311  tmp = 0;
312  }
313  localMaxpressureDiff = max(tmp, localMaxpressureDiff);
314  }
315  }
316  i = threadIdx.x;
317  sharedMaxpressureDiff[i] = localMaxpressureDiff;
318  for (j = NUM_THREADS_PC_INTEGRATOR / 2; j > 0; j /= 2) {
319  __syncthreads();
320  if (i < j) {
321  k = i + j;
322  sharedMaxpressureDiff[i] = localMaxpressureDiff = max(localMaxpressureDiff, sharedMaxpressureDiff[k]);
323  }
324  }
325  // write block result to global memory
326  if (i == 0) {
327  k = blockIdx.x;
328  maxpressureDiffPerBlock[k] = localMaxpressureDiff;
329  m = gridDim.x - 1;
330  if (m == atomicInc((unsigned int *)&blockCount, m)) {
331  // last block, so combine all block results
332  for (j = 0; j <= m; j++) {
333  localMaxpressureDiff = max(localMaxpressureDiff, maxpressureDiffPerBlock[j]);
334  }
335  maxpressureDiff = localMaxpressureDiff;
336  // reset block count
337  blockCount = 0;
338  }
339  if (maxpressureDiff > max_abs_pressure_change) {
340  printf("%e\n", maxpressureDiff);
341  pressureChangeSmallEnough = FALSE;
342  dt = 0.25 * dt;
343  dt = min(dt, endTimeD - currentTimeD);
344  } else {
345  pressureChangeSmallEnough = TRUE;

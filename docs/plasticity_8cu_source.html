37  register double alpha_phi, kc, I1, sqrt_J2, rn;
38 #if DIM == 2
39  register double shear, bulk, poissons_ratio, sz;
40 #endif
41  register double S_i[DIM][DIM];
42  inc = blockDim.x * gridDim.x;
43  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
44  matId = p_rhs.materialId[i];
45  if (EOS_TYPE_REGOLITH == matEOS[matId]) {
46 
47  alpha_phi = matAlphaPhi[matId];
48  kc = matCohesionCoefficient[matId];
49 
50 #if DIM == 2
51  shear = matShearmodulus[matId];
52  bulk = matBulkmodulus[matId];
53  poissons_ratio = (3*bulk - 2*shear) / (2*(3*bulk + shear));
54  I1 = (1 + poissons_ratio) * (p.S[stressIndex(i, 0, 0)] + p.S[stressIndex(i, 1, 1)]);
55 #else
56  I1 = p.S[stressIndex(i,0,0)] + p.S[stressIndex(i,1,1)] + p.S[stressIndex(i,2,2)];
57 #endif
58 
59  //Tension cracking treatment
60  //Equation 29, Bui et al., 2008
61  if (-I1*alpha_phi + kc < 0) {
62  for (d = 0; d < DIM; d++) {
63  p.S[stressIndex(i, d, d)] -= (I1 - kc/alpha_phi)/3.0;
64  }
65  }
66 
67 #if DIM == 2
68  I1 = (1 + poissons_ratio) * (p.S[stressIndex(i, 0, 0)] + p.S[stressIndex(i, 1, 1)]);
69 #else
70  I1 = p.S[stressIndex(i,0,0)] + p.S[stressIndex(i,1,1)] + p.S[stressIndex(i,2,2)];
71 #endif
72 
73  //get S
74  for (d = 0; d < DIM; d++) {
75  for (e = 0; e < DIM; e++) {
76  S_i[d][e] = p.S[stressIndex(i, d, e)];
77  }
78  S_i[d][d] -= I1/3.0;
79  }
80 #if DIM == 2
81  sz = poissons_ratio*(S_i[0][0] + S_i[1][1]);
82 #endif
83 
84  sqrt_J2 = 0.0;
85  for (d = 0; d < DIM; d++) {
86  for (e = 0; e < DIM; e++) {
87  sqrt_J2 += S_i[d][e]*S_i[d][e];
88  }
89  }
90 #if DIM == 2
91  sqrt_J2 += sz*sz;
92 #endif
93  sqrt_J2 *= 0.5;
94  sqrt_J2 = sqrt(sqrt_J2);
95 
96  //stress-scaling
97  //Equation 31, Bui et al., 2008
98  if (sqrt_J2 > 0) {
99  rn = (-I1*alpha_phi + kc) / sqrt_J2;
100  rn = min(rn, 1.0);
101  for (d = 0; d < DIM; d++) {
102  for (e = 0; e < DIM; e++) {
103  p.S[stressIndex(i, d, e)] = rn*S_i[d][e];
104  }
105  p.S[stressIndex(i, d, d)] += I1/3.0;
106  }
107  }
108  } //end if (EOS_TYPE_REGOLITH)
109  }
110 }

116 __global__ void plasticityModel(void) {
117  // introduce plastic behaviour by limiting the deviatoric stress
118  register int i, inc, d, e;
119  register double mises_f, tmp;
120  register double I1, J2, sqrt_J2;
121  register double y, y_i, y_d, y_M, y_0, y_0_d, damage, e_melt;
122  /* drucker prager constants */
123  register double A, B;
124  double mu_i, mu_d; // coefficients of internal friction
125 
126  inc = blockDim.x * gridDim.x;
127  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
128 
129  // VISCOUS_REGOLITH is treated in timeintegration.cu when \sigma is calculated
130  if (matEOS[p_rhs.materialId[i]] == EOS_TYPE_VISCOUS_REGOLITH) {
131  continue;
132  }
133 
134  mises_f = 1.0;
135 
136  /* second invariant of the deviator stress tensor */
137  J2 = 0.0;
138  for (d = 0; d < DIM; d++) {
139  for (e = 0; e < DIM; e++) {
140  tmp = p.S[stressIndex(i, d, e)];
141  J2 += tmp*tmp;
142  }
143  }
144  J2 *= 0.5;
145  sqrt_J2 = sqrt(J2);
146 
147  /* first invariant of the stress tensor */
148  I1 = -3.0 * p.p[i];
149 
150 #if MOHR_COULOMB_PLASTICITY
151  // Mohr-Coulomb yield criterion
152  // matInternalFriction = \mu = tan(matFrictionAngle)
153  y = matCohesion[p_rhs.materialId[i]] + matInternalFriction[p_rhs.materialId[i]] * p.p[i];
154 
155  // additional von Mises limit if set
156 # if VON_MISES_PLASTICITY
157  y = min(y, matYieldStress[p_rhs.materialId[i]]);
158 # endif
159  if (y < 0.0) {
160  y = 0.0;
161  }
162 
163  // Drucker-Prager-like -> compare to sqrt(J2)
164  if (J2 > 0.0) {
165  mises_f = y/sqrt_J2;
166  }
167 #elif DRUCKER_PRAGER_PLASTICITY
168  A = B = 0;
169  // Drucker-Prager constants from Mohr-Coulomb constants -> 3D!
170  A = 6. * matCohesion[p_rhs.materialId[i]] * cos(matFrictionAngle[p_rhs.materialId[i]])
171  / (sqrt(3.) * (3. - sin(matFrictionAngle[p_rhs.materialId[i]])));
172  B = 2. * sin(matFrictionAngle[p_rhs.materialId[i]]) / (sqrt(3.) * (3. - sin(matFrictionAngle[p_rhs.materialId[i]])));
173 
174  // yield strength determined by Drucker-Prager condition
175  y = A + 3.0*p.p[i]*B;
176 
177  // additional von Mises limit if set
178 # if VON_MISES_PLASTICITY
179  y = min(y, matYieldStress[p_rhs.materialId[i]]);
180 # endif
181  if (y < 0.0) {
182  y = 0.0;
183  }
184 
185  // Drucker-Prager-like -> compare to sqrt(J2)
186  if (J2 > 0.0) {
187  mises_f = y/sqrt_J2;
188  }
189 #elif COLLINS_PLASTICITY
190  y_0 = matCohesion[p_rhs.materialId[i]];
191  y_M = matYieldStress[p_rhs.materialId[i]];
192  mu_i = matInternalFriction[p_rhs.materialId[i]];
193 
194  // yield strength of intact material, with constant cohesion for p<0
195  y_i = y_0;
196  if (p.p[i] > 0.0) {
197  y_i += mu_i * p.p[i]
198  / (1 + mu_i * p.p[i] / (y_M - y_0) );
199  }
200 # if COLLINS_PLASTICITY_INCLUDE_MELT_ENERGY
201  e_melt = matMeltEnergy[p_rhs.materialId[i]];
202 
203  if (p.e[i] >= e_melt) {
204  y_i = 0.0;
205  } else if (p.e[i] > 0.0) {
206  y_i *= ( 1.0 - p.e[i] / e_melt );
207  }
208 # endif
209 # if FRAGMENTATION
210  y_0_d = matCohesionDamaged[p_rhs.materialId[i]];
211  mu_d = matInternalFrictionDamaged[p_rhs.materialId[i]];
212  damage = p.damage_total[i];
213  if (damage > 1.0) damage = 1.0;
214  if (damage < 0.0) damage = 0.0;
215 
216  // yield strength of damaged material, with the cohesion going (linearly) to zero for p<0
217  y_d = y_0_d + mu_d * p.p[i];
218  if (y_d < 0.0)
219  y_d = 0.0;
220 
221  // the actual yield strength Y is a weighted mean of Y_i and Y_d
222  // note: therefore potential melt-energy effects are also included in Y
223  y = (1.0-damage) * y_i + damage * y_d;
224 
225  // always limit the yield strength to the intact value
226  if (y > y_i)
227  y = y_i;
228 # else
229  y = y_i;
230 # endif
231  // Drucker-Prager-like -> compare to sqrt(J2)
232  if (J2 > 0.0) {
233  mises_f = y/sqrt_J2;
234  }
235 #elif COLLINS_PLASTICITY_SIMPLE
236  y_0 = matCohesion[p_rhs.materialId[i]];
237  y_M = matYieldStress[p_rhs.materialId[i]];
238  mu_i = matInternalFriction[p_rhs.materialId[i]];
239 
240  // unlike for the regular Collins model, here we let the yield strength decrease to zero for p<0,
241  // following a linear decline for p<0, where the slope is mu_i, and the zero at -y_0/mu_i
242  if( p.p[i] > 0.0 ) {
243  y = y_0 + mu_i * p.p[i]
244  / (1.0 + mu_i * p.p[i] / (y_M - y_0) );
245  } else if( p.p[i] > -y_0 / mu_i ) {
246  y = y_0 + p.p[i] * mu_i;
247  } else {
248  y = 0.0;
249  }
250 
251  // let the yield strength decrease to zero for p<0 following the regular Y_i curve,
252  // where the zero is at p_0 = -Y_0 (Y_M-Y_0) / (mu_i Y_M)
253 // if ( p.p[i] > y_0*(y_0-y_M)/(mu_i*y_M) ) {
254 // y = y_0 + mu_i * p.p[i]
255 // / (1.0 + mu_i * p.p[i] / (y_M - y_0) );
256 // } else {
257 // y = 0.0;
258 // }
259 
260  // also limit negative pressures to value at zero of yield strength curve
261  // (zero is at -y_0/mu_i if assumed linear for p<0)
262  if( p.p[i] < -y_0 / mu_i)
263  p.p[i] = -y_0 / mu_i;
264 
265  // Drucker-Prager-like -> compare to sqrt(J2)
266  if (J2 > 0.0) {
267  mises_f = y/sqrt_J2;
268  }
269 #else // simple von Mises yield criterion without *any* dependency
270  y = matYieldStress[p_rhs.materialId[i]];
271 # if SIRONO_POROSITY
272  // shear strength using Sironos model
273  if (matEOS[p_rhs.materialId[i]] == EOS_TYPE_SIRONO) {
274  y = sqrt((-1.0) * p.tensile_strength[i] * p.compressive_strength[i]);
275  p.shear_strength[i] = y;
276  } else {
277  p.shear_strength[i] = DBL_MAX;
278  y = p.shear_strength[i];
279  }
280 # endif
281  // von Mises limit like
282  if (J2 > 0.0) {
283  mises_f = y*y/(3.0*J2);
284  }
285 #endif
286  // finally limit the deviatoric stress tensor
287  if (mises_f > 1.0)
288  mises_f = 1.0;
289  for (d = 0; d < DIM; d++) {
290  for (e = 0; e < DIM; e++) {
291  p.S[stressIndex(i, d, e)] *= mises_f;
292  }
293  }
294  }
295 }
296 #endif

302  // introduce plastic behaviour by limiting the deviatoric stress
303  register int i, inc, d, e;
304  register double J2, jc_f, y_0, tmp;
305  register double y_jc = 0;
306  register double T_star = 0;
307  register double B, n, m, edot0, C, Tref, Tmelt;
308  /*register double Cp, CV;*/
309 
310  inc = blockDim.x * gridDim.x;
311  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
312 
313  J2 = 0;
314  jc_f = 0;
315  for (d = 0; d < DIM; d++) {
316  for (e = 0; e < DIM; e++) {
317  tmp = p.S[stressIndex(i, d, e)];
318  J2 += tmp*tmp;
319  }
320  }
321 
322  y_0 = matjc_y0[p_rhs.materialId[i]];
323  B = matjc_B[p_rhs.materialId[i]];
324  n = matjc_n[p_rhs.materialId[i]];
325  m = matjc_m[p_rhs.materialId[i]];
326  edot0 = matjc_edot0[p_rhs.materialId[i]];
327  C = matjc_C[p_rhs.materialId[i]];
328  Tref = matjc_Tref[p_rhs.materialId[i]];
329  Tmelt = matjc_Tmelt[p_rhs.materialId[i]];
330  /*Cp = matCp[p_rhs.materialId[i]];*/
331  /*CV = matCV[p_rhs.materialId[i]];*/
332 
333  register double edotp = p.edotp[i];
334  register double ep = p.ep[i];
335  register double T = p.T[i];
336 
337  // T_star has to be different for different cases, otherwise we have complex numbers and nans
338  if (T < Tref) {
339  T_star = 0;
340  } else if (T > Tmelt) {
341  T_star = 1;
342  } else {
343  T_star = (T - Tref) / (Tmelt - Tref);
344  }
345 
346  // Calculating flow stress according to Johnson and Cook
347  if (edotp > 0) {
348  y_jc = (y_0 + B*(pow(ep,n))) * (1 + C*log(edotp / edot0)) * (1 - pow(T_star,m));
349  } else {
350  y_jc = y_0;
351  }
352 
353  y_jc = y_jc * y_jc;
354  J2 = J2 * 1.5;
355  if (J2 > y_jc)
356  jc_f = y_jc/J2;
357  else
358  jc_f = 1;
359  for (d = 0; d < DIM; d++) {
360  for (e = 0; e < DIM; e++) {
361  p.S[stressIndex(i, d, e)] *= jc_f;
362  }
363  }
364 
365  /* remember for calculation of edotp later on */
366  p.jc_f[i] = jc_f;
367  p.edotp[i] = 0.0;
368  }
369 }

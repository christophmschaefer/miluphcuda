33 __global__ void plasticity()
34 {
35  register int i, inc, matId, d, e;
36  register double alpha_phi, kc, I1, sqrt_J2, rn;
37 #if DIM == 2
38  register double shear, bulk, poissons_ratio, sz;
39 #endif
40  register double S_i[DIM][DIM];
41  inc = blockDim.x * gridDim.x;
42  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
43  matId = p_rhs.materialId[i];
44  if (EOS_TYPE_REGOLITH == matEOS[matId]) {
45 
46  alpha_phi = matAlphaPhi[matId];
47  kc = matCohesionCoefficient[matId];
48 
49 
50 #if DIM == 2
51  shear = matShearmodulus[matId];
52  bulk = matBulkmodulus[matId];
53  poissons_ratio = (3*bulk - 2*shear) / (2*(3*bulk + shear));
54  I1 = (1 + poissons_ratio) * (p.S[stressIndex(i, 0, 0)] + p.S[stressIndex(i, 1, 1)]);
55 #else
56  I1 = p.S[stressIndex(i,0,0)] + p.S[stressIndex(i,1,1)] + p.S[stressIndex(i,2,2)];
57 #endif
58 
59 
60  //Tension cracking treatment
61  //Equation 29, Bui et al., 2008
62  if (-I1*alpha_phi + kc < 0) {
63  for (d = 0; d < DIM; d++) {
64  p.S[stressIndex(i, d, d)] -= (I1 - kc/alpha_phi)/3.0;
65  }
66  }
67 
68 #if DIM == 2
69  I1 = (1 + poissons_ratio) * (p.S[stressIndex(i, 0, 0)] + p.S[stressIndex(i, 1, 1)]);
70 #else
71  I1 = p.S[stressIndex(i,0,0)] + p.S[stressIndex(i,1,1)] + p.S[stressIndex(i,2,2)];
72 #endif
73 
74 
75  //get S
76  for (d = 0; d < DIM; d++) {
77  for (e = 0; e < DIM; e++) {
78  S_i[d][e] = p.S[stressIndex(i, d, e)];
79  }
80  S_i[d][d] -= I1/3.0;
81  }
82 #if DIM == 2
83  sz = poissons_ratio*(S_i[0][0] + S_i[1][1]);
84 #endif
85 
86  sqrt_J2 = 0.0;
87  for (d = 0; d < DIM; d++) {
88  for (e = 0; e < DIM; e++) {
89  sqrt_J2 += S_i[d][e]*S_i[d][e];
90  }
91  }
92 #if DIM == 2
93  sqrt_J2 += sz*sz;
94 #endif
95  sqrt_J2 *= 0.5;
96  sqrt_J2 = sqrt(sqrt_J2);
97 
98 
99  //stress-scaling
100  //Equation 31, Bui et al., 2008
101  if (sqrt_J2 > 0) {
102  rn = (-I1*alpha_phi + kc) / sqrt_J2;
103  rn = min(rn, 1.0);
104  for (d = 0; d < DIM; d++) {
105  for (e = 0; e < DIM; e++) {
106  p.S[stressIndex(i, d, e)] = rn*S_i[d][e];
107  }
108  p.S[stressIndex(i, d, d)] += I1/3.0;
109  }
110  }
111 
112 
113  } //end if (EOS_TYPE_REGOLITH)
114  }
115 }
116 #endif

119 __global__ void vonMisesPlasticity(void) {
120  // introduce plastic behaviour by limiting the deviatoric stress
121  register int i, inc, d, e;
122  register double mises_f, tmp;
123  register double I1, J2, sqrt_J2;
124  register double y, y_i, y_M, y_0, ytmp;
125  register double damage;
126  /* drucker prager constants */
127  register double A, B;
128  double mu_i, mu_d; // coefficients of internal friction
129 
130  inc = blockDim.x * gridDim.x;
131  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
132  // VISCOUS_REGOLITH is treated in timeintegration.cu when \sigma is calculated
133  if (matEOS[p_rhs.materialId[i]] == EOS_TYPE_VISCOUS_REGOLITH) {
134  continue;
135  }
136 
137  /* second invariant of the stress tensor */
138  J2 = 0;
139  mises_f = 1;
140 
141  for (d = 0; d < DIM; d++) {
142  for (e = 0; e < DIM; e++) {
143  tmp = p.S[stressIndex(i, d, e)];
144  J2 += tmp*tmp;
145  }
146  }
147 
148  J2 *= 0.5;
149  sqrt_J2 = sqrt(J2);
150 
151  /* first invariant of the stress tensor */
152  I1 = -3.0 * p.p[i];
153 
154 
155 #if MOHR_COULOMB_PLASTICITY
156  // mohr coulomb yield criterion
157  // matInternalFriction = \mu = tan(matFrictionAngle)
158  y = matCohesion[p_rhs.materialId[i]];
159  if (p.p[i] > 0) {
160  y += matInternalFriction[p_rhs.materialId[i]] * p.p[i];
161  }
162  // drucker prager like -> compare to sqrt(J2)
163  if (J2 > 0) {
164  mises_f = y/sqrt_J2;
165  }
166 
167  if (mises_f > 1)
168  mises_f = 1;
169 #elif DRUCKER_PRAGER_PLASTICITY
170  A = B = 0;
171  // drucker prager constants from mohr-coulomb constants -> 3D!
172  A = 6. * matCohesion[p_rhs.materialId[i]] * cos(matFrictionAngle[p_rhs.materialId[i]])
173  / (sqrt(3.) * (3. - sin(matFrictionAngle[p_rhs.materialId[i]])));
174  B = 2. * sin(matFrictionAngle[p_rhs.materialId[i]]) / (sqrt(3.) * (3. - sin(matFrictionAngle[p_rhs.materialId[i]])));
175 
176  // yield strength determined by drucker prager condition
177  y = A;
178  if (p.p[i] > 0) {
179  y += 3.0*p.p[i]*B;
180  }
181  // drucker prager like -> compare to sqrt(J2)
182  if (J2 > 0) {
183  mises_f = y/sqrt_J2;
184  }
185 
186  if (mises_f > 1)
187  mises_f = 1;
188 #elif COLLINS_PRESSURE_DEPENDENT_YIELD_STRENGTH
189  y_0 = matCohesion[p_rhs.materialId[i]];
190  y_M = matYieldStress[p_rhs.materialId[i]];
191  mu_i = matInternalFriction[p_rhs.materialId[i]];
192 #if FRAGMENTATION
193  mu_d = matInternalFrictionDamaged[p_rhs.materialId[i]];
194 #endif
195  // shear strength of the intact material
196  ytmp = y_0;
197  if (p.p[i] > 0) {
198  ytmp += mu_i * p.p[i]
199  / (1 + mu_i * p.p[i] / (y_M - y_0) );
200  } else {
201  ytmp = y_0;
202  }
203 #if FRAGMENTATION
204  damage = p.damage_total[i];
205  if (damage > 1) damage = 1.0;
206  // yield strength of damaged material
207  if (p.p[i] > 0) {
208  y = mu_d * p.p[i];
209  /* limit the yield strength of damaged material to the yield strength of intact material */
210  if (damage < 1) {
211  y = (1-damage) * ytmp + damage*y;
212  if (y > ytmp) y = ytmp;
213  }
214  } else {
215  y = y_0;
216  }
217 #else
218  y = ytmp;
219 #endif
220  // drucker prager like -> compare to sqrt(J2)
221  if (J2 > 0) {
222  mises_f = y/sqrt_J2;
223  }
224 
225  if (mises_f > 1)
226  mises_f = 1;
227 #else // simple von Mises yield criterion without *any* dependency
228  y = matYieldStress[p_rhs.materialId[i]];
229 #if SIRONO_POROSITY
230  // Shear Strength using Sironos Model
231  if (matEOS[p_rhs.materialId[i]] == EOS_TYPE_SIRONO) {
232  y = sqrt((-1.0) * p.tensile_strength[i] * p.compressive_strength[i]);
233  p.shear_strength[i] = y;
234  } else {
235  p.shear_strength[i] = DBL_MAX;
236  y = p.shear_strength[i];
237  }
238 #endif
239  // von mises limit like
240  if (J2 > 0) {
241  mises_f = y*y/(3*J2);
242  }
243 
244  if (mises_f > 1)
245  mises_f = 1;
246 #endif
247 
248  for (d = 0; d < DIM; d++) {
249  for (e = 0; e < DIM; e++) {
250  p.S[stressIndex(i, d, e)] *= mises_f;
251  }
252  }
253  }
254 }
255 
256 #endif

263 #if JC_PLASTICITY
264 __global__ void JohnsonCookPlasticity(void) {
265  // introduce plastic behaviour by limiting the deviatoric stress
266  register int i, inc, d, e;
267  register double J2, jc_f, y_0, tmp;
268 
269  register double y_jc = 0;
270  register double T_star = 0;
271  register double B, n, m, edot0, C, Tref, Tmelt;
272  /*register double Cp, CV;*/
273 
274  inc = blockDim.x * gridDim.x;
275  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
276 
277  J2 = 0;
278  jc_f = 1;
279  for (d = 0; d < DIM; d++) {
280  for (e = 0; e < DIM; e++) {
281  tmp = p.S[stressIndex(i, d, e)];
282  J2 += tmp*tmp;
283  }
284  }
285 
286 
287  y_0 = matjc_y0[p_rhs.materialId[i]];
288  B = matjc_B[p_rhs.materialId[i]];
289  n = matjc_n[p_rhs.materialId[i]];
290  m = matjc_m[p_rhs.materialId[i]];
291  edot0 = matjc_edot0[p_rhs.materialId[i]];
292  C = matjc_C[p_rhs.materialId[i]];
293  Tref = matjc_Tref[p_rhs.materialId[i]];
294  Tmelt = matjc_Tmelt[p_rhs.materialId[i]];
295  /*Cp = matCp[p_rhs.materialId[i]];*/
296  /*CV = matCV[p_rhs.materialId[i]];*/
297 
298  register double edotp = p.edotp[i];
299  register double ep = p.ep[i];
300  register double T = p.T[i];
301 
302  // T_star has to be different for different cases, otherwise we have complex numbers and nans
303  if (T < Tref) {
304  T_star = 0;
305  } else if (T > Tmelt) {
306  T_star = 1;
307  } else {
308  T_star = (T - Tref) / (Tmelt - Tref);
309  }
310 
311 
312  // Calculating flow stress according to Johnson and Cook
313  if (edotp > 0) {
314  y_jc = (y_0 + B*(pow(ep,n))) * (1 + C*log(edotp / edot0)) * (1 - pow(T_star,m));
315  } else {
316  y_jc = y_0;
317  }
318 
319 
320  y_jc = y_jc * y_jc;
321  J2 = J2 * 1.5;
322  if (J2 > y_jc)
323  jc_f = y_jc/J2;
324  else
325  jc_f = 1;
326  for (d = 0; d < DIM; d++) {
327  for (e = 0; e < DIM; e++) {
328  p.S[stressIndex(i, d, e)] *= jc_f;
329  }
330  }
331 
332  /* remember for calculation of edotp later on */
333  p.jc_f[i] = jc_f;
334  p.edotp[i] = 0.0;
335  }
336 
337 }

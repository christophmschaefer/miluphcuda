33 #if SOLID
34 __global__ void plasticity()
35 {
36  register int i, inc, matId, d, e;
37  register double alpha_phi, kc, I1, sqrt_J2, rn;
38 #if DIM == 2
39  register double shear, bulk, poissons_ratio, sz;
40 #endif
41  register double S_i[DIM][DIM];
42  inc = blockDim.x * gridDim.x;
43  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
44  matId = p_rhs.materialId[i];
45  if (EOS_TYPE_REGOLITH == matEOS[matId]) {
46 
47  alpha_phi = matAlphaPhi[matId];
48  kc = matCohesionCoefficient[matId];
49 
50 
51 #if DIM == 2
52  shear = matShearmodulus[matId];
53  bulk = matBulkmodulus[matId];
54  poissons_ratio = (3*bulk - 2*shear) / (2*(3*bulk + shear));
55  I1 = (1 + poissons_ratio) * (p.S[stressIndex(i, 0, 0)] + p.S[stressIndex(i, 1, 1)]);
56 #else
57  I1 = p.S[stressIndex(i,0,0)] + p.S[stressIndex(i,1,1)] + p.S[stressIndex(i,2,2)];
58 #endif
59 
60 
61  //Tension cracking treatment
62  //Equation 29, Bui et al., 2008
63  if (-I1*alpha_phi + kc < 0) {
64  for (d = 0; d < DIM; d++) {
65  p.S[stressIndex(i, d, d)] -= (I1 - kc/alpha_phi)/3.0;
66  }
67  }
68 
69 #if DIM == 2
70  I1 = (1 + poissons_ratio) * (p.S[stressIndex(i, 0, 0)] + p.S[stressIndex(i, 1, 1)]);
71 #else
72  I1 = p.S[stressIndex(i,0,0)] + p.S[stressIndex(i,1,1)] + p.S[stressIndex(i,2,2)];
73 #endif
74 
75 
76  //get S
77  for (d = 0; d < DIM; d++) {
78  for (e = 0; e < DIM; e++) {
79  S_i[d][e] = p.S[stressIndex(i, d, e)];
80  }
81  S_i[d][d] -= I1/3.0;
82  }
83 #if DIM == 2
84  sz = poissons_ratio*(S_i[0][0] + S_i[1][1]);
85 #endif
86 
87  sqrt_J2 = 0.0;
88  for (d = 0; d < DIM; d++) {
89  for (e = 0; e < DIM; e++) {
90  sqrt_J2 += S_i[d][e]*S_i[d][e];
91  }
92  }
93 #if DIM == 2
94  sqrt_J2 += sz*sz;
95 #endif
96  sqrt_J2 *= 0.5;
97  sqrt_J2 = sqrt(sqrt_J2);
98 
99 
100  //stress-scaling
101  //Equation 31, Bui et al., 2008
102  if (sqrt_J2 > 0) {
103  rn = (-I1*alpha_phi + kc) / sqrt_J2;
104  rn = min(rn, 1.0);
105  for (d = 0; d < DIM; d++) {
106  for (e = 0; e < DIM; e++) {
107  p.S[stressIndex(i, d, e)] = rn*S_i[d][e];
108  }
109  p.S[stressIndex(i, d, d)] += I1/3.0;
110  }
111  }
112 
113 
114  } //end if (EOS_TYPE_REGOLITH)
115  }
116 }
117 #endif

122 __global__ void plasticityModel(void) {
123  // introduce plastic behaviour by limiting the deviatoric stress
124  register int i, inc, d, e;
125  register double mises_f, tmp;
126  register double I1, J2, sqrt_J2;
127  register double y, y_i, y_d, y_M, y_0, y_0_d, damage, e_melt;
128  /* drucker prager constants */
129  register double A, B;
130  double mu_i, mu_d; // coefficients of internal friction
131 
132  inc = blockDim.x * gridDim.x;
133  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
134  // VISCOUS_REGOLITH is treated in timeintegration.cu when \sigma is calculated
135  if (matEOS[p_rhs.materialId[i]] == EOS_TYPE_VISCOUS_REGOLITH) {
136  continue;
137  }
138 
139  mises_f = 1.0;
140 
141  /* second invariant of the deviator stress tensor */
142  J2 = 0.0;
143 
144  for (d = 0; d < DIM; d++) {
145  for (e = 0; e < DIM; e++) {
146  tmp = p.S[stressIndex(i, d, e)];
147  J2 += tmp*tmp;
148  }
149  }
150 
151  J2 *= 0.5;
152  sqrt_J2 = sqrt(J2);
153 
154  /* first invariant of the stress tensor */
155  I1 = -3.0 * p.p[i];
156 
157 #if MOHR_COULOMB_PLASTICITY
158  // mohr coulomb yield criterion
159  // matInternalFriction = \mu = tan(matFrictionAngle)
160  y = matCohesion[p_rhs.materialId[i]] + matInternalFriction[p_rhs.materialId[i]] * p.p[i];
161  if (y < 0.0) {
162  y = 0.0;
163  }
164 
165  // drucker prager like -> compare to sqrt(J2)
166  if (J2 > 0) {
167  mises_f = y/sqrt_J2;
168  }
169 #elif DRUCKER_PRAGER_PLASTICITY
170  A = B = 0;
171  // drucker prager constants from mohr-coulomb constants -> 3D!
172  A = 6. * matCohesion[p_rhs.materialId[i]] * cos(matFrictionAngle[p_rhs.materialId[i]])
173  / (sqrt(3.) * (3. - sin(matFrictionAngle[p_rhs.materialId[i]])));
174  B = 2. * sin(matFrictionAngle[p_rhs.materialId[i]]) / (sqrt(3.) * (3. - sin(matFrictionAngle[p_rhs.materialId[i]])));
175 
176  // yield strength determined by drucker prager condition
177  y = A + 3.0*p.p[i]*B;
178  if (y < 0.0) {
179  y = 0.0;
180  }
181 
182  // drucker prager like -> compare to sqrt(J2)
183  if (J2 > 0) {
184  mises_f = y/sqrt_J2;
185  }
186 #elif COLLINS_PLASTICITY
187  y_0 = matCohesion[p_rhs.materialId[i]];
188  y_M = matYieldStress[p_rhs.materialId[i]];
189  mu_i = matInternalFriction[p_rhs.materialId[i]];
190 
191  // yield strength of intact material, with (constant) cohesion also for p<0
192  y_i = y_0;
193  if (p.p[i] > 0) {
194  y_i += mu_i * p.p[i]
195  / (1 + mu_i * p.p[i] / (y_M - y_0) );
196  }
197 # if COLLINS_PLASTICITY_INCLUDE_MELT_ENERGY
198  e_melt = matMeltEnergy[p_rhs.materialId[i]];
199 
200  if (p.e[i] >= e_melt) {
201  y_i = 0.0;
202  } else if (p.e[i] > 0.0) {
203  y_i *= ( 1.0 - p.e[i] / e_melt );
204  }
205 # endif
206 # if FRAGMENTATION
207  y_0_d = matCohesionDamaged[p_rhs.materialId[i]];
208  mu_d = matInternalFrictionDamaged[p_rhs.materialId[i]];
209  damage = p.damage_total[i];
210  if (damage > 1.0) damage = 1.0;
211  if (damage < 0.0) damage = 0.0;
212 
213  // yield strength of damaged material, with the cohesion going (linearly) to zero for p<0
214  y_d = y_0_d + mu_d * p.p[i];
215  if (y_d < 0.0)
216  y_d = 0.0;
217 
218  // the actual yield strength Y is a weighted mean of Y_i and Y_d
219  // (therefore potential melt-energy effects are also included in Y)
220  y = (1.0-damage) * y_i + damage * y_d;
221 
222  // always limit the yield strength to the intact value
223  if (y > y_i)
224  y = y_i;
225 # else
226  y = y_i;
227 # endif
228  // drucker prager like -> compare to sqrt(J2)
229  if (J2 > 0) {
230  mises_f = y/sqrt_J2;
231  }
232 #elif COLLINS_PLASTICITY_SIMPLE
233  y_0 = matCohesion[p_rhs.materialId[i]];
234  y_M = matYieldStress[p_rhs.materialId[i]];
235  mu_i = matInternalFriction[p_rhs.materialId[i]];
236 
237  // unlike for the regular Collins model, here we let the yield strength decrease to
238  // the first zero for p<0, and set it zero for even greater negative pressure
239  // the zero is at p_0 = -Y_0 (Y_M-Y_0) / (mu_i Y_M)
240  if ( p.p[i] > y_0*(y_0-y_M)/(mu_i*y_M) ) {
241  y = y_0 + mu_i * p.p[i]
242  / (1 + mu_i * p.p[i] / (y_M - y_0) );
243  } else {
244  y = 0.0;
245  }
246 
247  // drucker prager like -> compare to sqrt(J2)
248  if (J2 > 0) {
249  mises_f = y/sqrt_J2;
250  }
251 #else // simple von Mises yield criterion without *any* dependency
252  y = matYieldStress[p_rhs.materialId[i]];
253 # if SIRONO_POROSITY
254  // Shear Strength using Sironos Model
255  if (matEOS[p_rhs.materialId[i]] == EOS_TYPE_SIRONO) {
256  y = sqrt((-1.0) * p.tensile_strength[i] * p.compressive_strength[i]);
257  p.shear_strength[i] = y;
258  } else {
259  p.shear_strength[i] = DBL_MAX;
260  y = p.shear_strength[i];
261  }
262 # endif
263  // von Mises limit like
264  if (J2 > 0) {
265  mises_f = y*y/(3*J2);
266  }
267 #endif
268  // finally limit the deviatoric stress tensor
269  if (mises_f > 1.0)
270  mises_f = 1.0;
271  for (d = 0; d < DIM; d++) {
272  for (e = 0; e < DIM; e++) {
273  p.S[stressIndex(i, d, e)] *= mises_f;
274  }
275  }
276  }
277 }
278 #endif

283 #if JC_PLASTICITY
284 __global__ void JohnsonCookPlasticity(void) {
285  // introduce plastic behaviour by limiting the deviatoric stress
286  register int i, inc, d, e;
287  register double J2, jc_f, y_0, tmp;
288 
289  register double y_jc = 0;
290  register double T_star = 0;
291  register double B, n, m, edot0, C, Tref, Tmelt;
292  /*register double Cp, CV;*/
293 
294  inc = blockDim.x * gridDim.x;
295  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
296 
297  J2 = 0;
298  jc_f = 0;
299  for (d = 0; d < DIM; d++) {
300  for (e = 0; e < DIM; e++) {
301  tmp = p.S[stressIndex(i, d, e)];
302  J2 += tmp*tmp;
303  }
304  }
305 
306 
307  y_0 = matjc_y0[p_rhs.materialId[i]];
308  B = matjc_B[p_rhs.materialId[i]];
309  n = matjc_n[p_rhs.materialId[i]];
310  m = matjc_m[p_rhs.materialId[i]];
311  edot0 = matjc_edot0[p_rhs.materialId[i]];
312  C = matjc_C[p_rhs.materialId[i]];
313  Tref = matjc_Tref[p_rhs.materialId[i]];
314  Tmelt = matjc_Tmelt[p_rhs.materialId[i]];
315  /*Cp = matCp[p_rhs.materialId[i]];*/
316  /*CV = matCV[p_rhs.materialId[i]];*/
317 
318  register double edotp = p.edotp[i];
319  register double ep = p.ep[i];
320  register double T = p.T[i];
321 
322  // T_star has to be different for different cases, otherwise we have complex numbers and nans
323  if (T < Tref) {
324  T_star = 0;
325  } else if (T > Tmelt) {
326  T_star = 1;
327  } else {
328  T_star = (T - Tref) / (Tmelt - Tref);
329  }
330 
331 
332  // Calculating flow stress according to Johnson and Cook
333  if (edotp > 0) {
334  y_jc = (y_0 + B*(pow(ep,n))) * (1 + C*log(edotp / edot0)) * (1 - pow(T_star,m));
335  } else {
336  y_jc = y_0;
337  }
338 
339 
340  y_jc = y_jc * y_jc;
341  J2 = J2 * 1.5;
342  if (J2 > y_jc)
343  jc_f = y_jc/J2;
344  else
345  jc_f = 1;
346  for (d = 0; d < DIM; d++) {
347  for (e = 0; e < DIM; e++) {
348  p.S[stressIndex(i, d, e)] *= jc_f;
349  }
350  }
351 
352  /* remember for calculation of edotp later on */
353  p.jc_f[i] = jc_f;
354  p.edotp[i] = 0.0;
355  }
356 
357 }

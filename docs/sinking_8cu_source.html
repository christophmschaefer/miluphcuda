31 //function adds to sink: mass and velocity (angular momentum, linear momentum) of the accreted particle and calculate the new position of sink (COM)
32 __device__ void UpdateSinkValues(int sink_num, int particle_id)
33 {
34  //COM - position - velocity
35  pointmass.x[sink_num] = (pointmass.m[sink_num]*pointmass.x[sink_num] + p.m[particle_id]*p.x[particle_id]) / (pointmass.m[sink_num] + p.m[particle_id]);
36  pointmass.y[sink_num] = (pointmass.m[sink_num]*pointmass.y[sink_num] + p.m[particle_id]*p.y[particle_id]) / (pointmass.m[sink_num] + p.m[particle_id]);
37 
38  pointmass.vx[sink_num] = (p.m[particle_id]*p.vx[particle_id] + pointmass.m[sink_num]*pointmass.vx[sink_num]) / (p.m[particle_id] + pointmass.m[sink_num]);
39  pointmass.vy[sink_num] = (p.m[particle_id]*p.vy[particle_id] + pointmass.m[sink_num]*pointmass.vy[sink_num]) / (p.m[particle_id] + pointmass.m[sink_num]);
40 
41 #if DIM == 3
42  pointmass.z[sink_num] = (pointmass.m[sink_num]*pointmass.z[sink_num] + p.m[particle_id]*p.z[particle_id]) / (pointmass.m[sink_num] + p.m[particle_id]);
43 
44  pointmass.vz[sink_num] = (p.m[particle_id]*p.vz[particle_id] + pointmass.m[sink_num]*pointmass.vz[sink_num]) / (p.m[particle_id] + pointmass.m[sink_num]);
45 
46 #endif //DIM == 3
47 
48  pointmass.m[sink_num] += p.m[particle_id];
49 }
50 #endif //UPDATE_SINK_VALUES

52 //function checks if particle is to be accreted on sink particle
53 __global__ void ParticleSinking()
54 {
55  register int i, inc, n;
56  double vel_esc, vel, dist_0, dist_1, distance, h, h_circ, r_x, r_y, r_z, v_x, v_y, v_z, h_x, h_y, h_z;
57  inc = blockDim.x * gridDim.x;
58  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
59 
60  //look for a particle with material type = -2 and reset it to material type = -1 if it is bound to sink particle
61  if (p_rhs.materialId[i] == EOS_TYPE_ACCRETED) {
62 
63  //distance between particle and each sink and particle velocity
64 #if DIM == 2
65  dist_0 = sqrt( (pointmass.x[0] - p.x[i])*(pointmass.x[0] - p.x[i]) + (pointmass.y[0] - p.y[i])*(pointmass.y[0] - p.y[i]) );
66  dist_1 = sqrt( (pointmass.x[1] - p.x[i])*(pointmass.x[1] - p.x[i]) + (pointmass.y[1] - p.y[i])*(pointmass.y[1] - p.y[i]) );
67 
68  vel = sqrt(p.vx[i]*p.vx[i] + p.vy[i]*p.vy[i]);
69 #endif //end DIM == 2
70 
71 #if DIM == 3
72  dist_0 = sqrt( (pointmass.x[0] - p.x[i])*(pointmass.x[0] - p.x[i]) + (pointmass.y[0] - p.y[i])*(pointmass.y[0] - p.y[i]) + (pointmass.z[0] - p.z[i])*(pointmass.z[0] - p.z[i]) );
73  dist_1 = sqrt( (pointmass.x[1] - p.x[i])*(pointmass.x[1] - p.x[i]) + (pointmass.y[1] - p.y[i])*(pointmass.y[1] - p.y[i]) + (pointmass.z[1] - p.z[i])*(pointmass.z[1] - p.z[i]) );
74 
75  vel = sqrt(p.vx[i]*p.vx[i] + p.vy[i]*p.vy[i] + p.vz[i]*p.vz[i]);
76 #endif //end DIM == 3
77 
78  if (dist_0 < dist_1) {
79  n = 0;
80  distance = dist_0;
81  }
82  else {
83  n = 1;
84  distance = dist_1;
85  }
86 
87  //escape velocity at r_acc(rmin)
88  vel_esc = sqrt(2 * C_GRAVITY * pointmass.m[n] / distance);
89 
90  //specific angular momentum of each particle about the sink particle
91  r_x = pointmass.x[n] - p.x[i];
92  r_y = pointmass.y[n] - p.y[i];
93 
94  v_x = pointmass.vx[n] - p.vx[i];
95  v_y = pointmass.vy[n] - p.vy[i];
96 
97  h_z = r_x*v_y - r_y*v_x;
98 
99 #if DIM == 2
100  h = sqrt(h_z*h_z);
101 #endif //DIM ==2
102 
103 #if DIM == 3
104  r_z = pointmass.z[n] - p.z[i];
105 
106  v_z = pointmass.vz[n] - p.vz[i];
107 
108  h_x = r_y*v_z - r_z*v_y;
109  h_y = r_z*v_x - r_x*v_z;
110  h = sqrt(h_x*h_x + h_y*h_y + h_z*h_z);
111 #endif //DIM == 3
112 
113  //specific angular momentum to form circular orbit at semi-major axis
114  h_circ = sqrt(C_GRAVITY * pointmass.m[n] * distance);
115 
116  //check if particle is to be accreted (bound) to sink: particle velocity < escape velocity && specific angular momentum of particle < angular momentum to form circular orbit
117  if (vel < vel_esc && h < h_circ) {
118 #if UPDATE_SINK_VALUES
119  UpdateSinkValues(n, i);
120 #endif //UPDATE_SINK_VALUES
121  p_rhs.materialId[i] = EOS_TYPE_IGNORE;
122  }
123  else {
124  //particle is not accreted, particle material set to inital one
125  p_rhs.materialId[i] = p_rhs.materialId0[i];
126  }
127  }
128  }
129 }
130 #endif //PARTICLE_ACCRETION

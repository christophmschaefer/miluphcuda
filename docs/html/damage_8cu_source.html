29 #if FRAGMENTATION
30 __global__ void damageLimit(void) {
31  register int i, inc;
32  volatile int nof, noaf;
33  volatile double dmg, dmgMax;
34  inc = blockDim.x * gridDim.x;
35  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
36  if (EOS_TYPE_IGNORE == matEOS[p_rhs.materialId[i]] || p_rhs.materialId[i] == EOS_TYPE_IGNORE) {
37  continue;
38  }
39  dmg = p.d[i];
40  if (dmg < 0) dmg = 0;
41  noaf = p.numActiveFlaws[i];
42  dmgMax = 1;
43  if (noaf < 1 && dmg > 0) {
44  printf("Error, not possible: noaf: %d, dmg %e\n", noaf, dmg);
45  assert(0);
46  }
47  if (noaf > 0) {
48  nof = p.numFlaws[i];
49  dmgMax = pow( ((double)noaf) / ((double)nof) , 1./DIM);
50  }
51 
52  if (dmg > dmgMax) {
53  dmg = dmgMax;
54  } else if (dmg < 0) {
55  printf("ERROR: DAMAGE is negative \t \t");
56  printf("%e %e %e %d %d\n", p.x[i], p.y[i], p.d[i], p.numFlaws[i], p.numActiveFlaws[i]);
57  assert(0);
58  }
59  p.d[i] = dmg;
60 #if PALPHA_POROSITY
61  if (p.damage_porjutzi[i] > 1.0) {
62  p.damage_porjutzi[i] = 1.0;
63  } else if (p.damage_porjutzi[i] < 0) {
64  p.damage_porjutzi[i] = 0.0;
65  }
66  dmg = p.d[i] + p.damage_porjutzi[i];
67  if (dmg > 1) dmg = 1.0;
68 #endif
69  p.damage_total[i] = dmg;
70  }
71 }
72 #endif

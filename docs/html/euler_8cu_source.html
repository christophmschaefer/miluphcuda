37 __global__ void integrateEuler(void)
38 {
39  register int i, inc;
40  inc = blockDim.x * gridDim.x;
41 #if GRAVITATING_POINT_MASSES
42  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numPointmasses; i += inc) {
43  pointmass.x[i] += dt * pointmass.vx[i];
44 #if DIM > 1
45  pointmass.y[i] += dt * pointmass.vy[i];
46  pointmass.vy[i] += dt * pointmass.ay[i];
47 #if DIM == 3
48  pointmass.z[i] += dt * pointmass.vz[i];
49 #endif
50 #endif
51  pointmass.vx[i] += dt * pointmass.ax[i];
52 #if DIM == 3
53  pointmass.vz[i] += dt * pointmass.az[i];
54 #endif
55 
56  }
57 #endif
58  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
59 #if INTEGRATE_DENSITY
60  p.rho[i] += dt * p.drhodt[i];
61 #endif
62 #if INTEGRATE_ENERGY
63  p.e[i] += dt * p.dedt[i];
64 #endif
65 
66 #if PALPHA_POROSITY
67  p.alpha_jutzi[i] += dt * p.dalphadt[i];
68 #endif
69 
70 #if SIRONO_POROSITY
71  p.rho_0prime[i] = p.rho_0prime[i];
72  p.rho_c_plus[i] = p.rho_c_plus[i];
73  p.rho_c_minus[i] = p.rho_c_minus[i];
74  p.compressive_strength[i] = p.compressive_strength[i];
75  p.tensile_strength[i] = p.tensile_strength[i];
76  p.shear_strength[i] = p.shear_strength[i];
77  p.K[i] = p.K[i];
78  p.flag_rho_0prime[i] = p.flag_rho_0prime[i];
79  p.flag_plastic[i] = p.flag_plastic[i];
80 #endif
81 
82 #if INTEGRATE_SML
83  p.h[i] += dt * p.dhdt[i];
84 #endif
85 #if JC_PLASTICITY
86  p.ep[i] += dt * p.edotp[i];
87  p.T[i] += dt * p.dTdt[i];
88 #endif
89 #if INVISCID_SPH
90  p.beta[i] += dt * p.dbetadt[i];
91 #endif
92 #if SOLID
93 #if FRAGMENTATION
94  p.d[i] += dt * p.dddt[i];
95 #if PALPHA_POROSITY
96  p.damage_porjutzi[i] += dt * p.ddamage_porjutzidt[i];
97  p.pold[i] = p.p[i];
98 #endif
99 #endif
100  int k;
101  for (k = 0; k < DIM*DIM; k++) {
102  p.S[i*DIM*DIM+k] += dt * p.dSdt[i*DIM*DIM+k];
103  }
104 #endif
105  p.x[i] += dt * p.dxdt[i];
106 #if DIM > 1
107  p.y[i] += dt * p.dydt[i];
108  p.vy[i] += dt * p.ay[i];
109 #if DIM == 3
110  p.z[i] += dt * p.dzdt[i];
111 #endif
112 #endif
113  p.vx[i] += dt * p.ax[i];
114 #if DIM == 3
115  p.vz[i] += dt * p.az[i];
116 #endif
117  }

123 void euler()
124 {
125  // integrate
126  int lastTimestep = startTimestep + numberOfTimesteps;
127  int timestep;
128  int eulerstep;
129  double tmptimestep = param.maxtimestep;
130  double endTime = startTime;
131  currentTime = startTime;
132  cudaVerify(cudaMemcpyToSymbol(currentTimeD, &currentTime, sizeof(double)));
133  cudaVerify(cudaMemcpyToSymbol(dt, &tmptimestep, sizeof(double)));
134 
135  for (timestep = startTimestep; timestep < lastTimestep; timestep++) {
136  eulerstep = 0;
137  endTime += timePerStep;
138  cudaVerify(cudaMemcpyToSymbol(endTimeD, &endTime, sizeof(double)));
139  // checking for changes in angular momentum
140  if (param.angular_momentum_check > 0) {
141  double L_current = calculate_angular_momentum();
142  double L_change_relative;
143  if (L_ini > 0) {
144  L_change_relative = fabs((L_ini - L_current)/L_ini);
145  }
146  if (param.verbose) {
147  fprintf(stdout, "Checking angular momentum conservation.\n");
148  fprintf(stdout, "Initial angular momentum: %.17e\n", L_ini);
149  fprintf(stdout, "Current angular momentum: %.17e\n", L_current);
150  fprintf(stdout, "Relative change: %.17e\n", L_change_relative);
151  }
152  if (L_change_relative > param.angular_momentum_check) {
153  fprintf(stderr, "Conservation of angular momentum violated. Exiting.\n");
154  exit(111);
155  }
156  }
157  while (currentTime < endTime) {
158  fprintf(stdout, "Euler Step # %d\n", ++eulerstep);
159  fprintf(stdout, " currenttime: %e \t endtime: %e, integrating with euler dt: %g\n", currentTime, endTime, param.maxtimestep);
160  rightHandSide();
161  if (currentTime + param.maxtimestep > endTime) {
162  tmptimestep = endTime - currentTime;
163  cudaVerify(cudaMemcpyToSymbol(dt, &tmptimestep, sizeof(double)));
164  currentTime += tmptimestep;
165  } else {
166  cudaVerify(cudaMemcpyToSymbol(dt, &param.maxtimestep, sizeof(double)));
167  currentTime += param.maxtimestep;
168  }
169  cudaVerifyKernel((integrateEuler<<<numberOfMultiprocessors, NUM_THREADS_EULER_INTEGRATOR>>>()));
170  //step was successful --> do something (e.g. look for min/max pressure...)
171  afterIntegrationStep();
172  }
173 
174  copyToHostAndWriteToFile(timestep, lastTimestep);
175  }
176 }

36 inline int _ConvertSMVer2Cores(int major, int minor)
37 /*!
38  * Helper function to calculate the number of CUDA core.
39  * Taken from cuda_samples/common/inc/helper_cuda.h
40  *********************************************************************** */
41 {
42  /* Defines for GPU Architecture types (using the SM version to determine the # of cores per SM */
43  typedef struct
44  {
45  int SM; /* 0xMm (hexidecimal notation), M = SM Major version, and m = SM minor version */
46  int Cores;
47  } sSMtoCores;
48 
49  sSMtoCores nGpuArchCoresPerSM[] =
50  {
51  { 0x10, 8 }, /* Tesla Generation (SM 1.0) G80 class */
52  { 0x11, 8 }, /* Tesla Generation (SM 1.1) G8x class */
53  { 0x12, 8 }, /* Tesla Generation (SM 1.2) G9x class */
54  { 0x13, 8 }, /* Tesla Generation (SM 1.3) GT200 class */
55  { 0x20, 32 }, /* Fermi Generation (SM 2.0) GF100 class */
56  { 0x21, 48 }, /* Fermi Generation (SM 2.1) GF10x class */
57  { 0x30, 192}, /* Kepler Generation (SM 3.0) GK10x class */
58  { 0x32, 192}, /* Kepler Generation (SM 3.2) GK10x class */
59  { 0x35, 192}, /* Kepler Generation (SM 3.5) GK11x class */
60  { 0x37, 192}, /* Kepler Generation (SM 3.7) GK21x class */
61  { 0x50, 128}, /* Maxwell Generation (SM 5.0) GM10x class */
62  { -1, -1 }
63  };
64 
65  int index = 0;
66 
67  while (nGpuArchCoresPerSM[index].SM != -1)
68  {
69  if (nGpuArchCoresPerSM[index].SM == ((major << 4) + minor))
70  {
71  return nGpuArchCoresPerSM[index].Cores;
72  }
73 
74  index++;
75  }
76 
77  /* If we don't find the values, we default use the previous one to run properly */
78  printf("MapSMtoCores for SM %d.%d is undefined. Default to use %d Cores/SM\n", major, minor, nGpuArchCoresPerSM[index-1].Cores);
79  return nGpuArchCoresPerSM[index-1].Cores;
80 }

84 void printfDeviceInformation(void)
85 /*!
86  * printfs some basic information about detected CUDA devices.
87  * Taken from cuda samples/1_Utilities/deviceQuery
88  *
89  *********************************************************************** */
90 {
91 int i, device_count, driverVersion = 0, runtimeVersion = 0;
92 struct cudaDeviceProp prop;
93 
94 cudaGetDeviceCount(&device_count);
95 
96  if (device_count == 0)
97  {
98  printf("\n> No device(s) that support CUDA found!\n");
99  exit(1);
100  }
101 
102  for (i = 0; i < device_count; i++)
103  {
104  // cudaSetDevice(i);
105  cudaGetDeviceProperties(&prop, i);
106  cudaDriverGetVersion(&driverVersion);
107  cudaRuntimeGetVersion(&runtimeVersion);
108 
109  printf("\n> General Information for Device %d -- %s\n\n", i, prop.name);
110  printf(" CUDA Driver Version: %d.%d\n", driverVersion/1000, (driverVersion%100)/10);
111  printf(" CUDA Runtime Version: %d.%d\n", runtimeVersion/1000,
112 (runtimeVersion%100)/10);
113  printf(" Compute capability: %d.%d\n\n", prop.major, prop.minor);
114 
115 
116  printf(" Multiprocessors: %d\n", prop.multiProcessorCount);
117  printf(" CUDA Cores / Multiprocessor: %d\n", _ConvertSMVer2Cores(prop.major, prop.minor));
118  printf(" Total amount of CUDA Cores: %d\n", _ConvertSMVer2Cores(prop.major,
119 prop.minor)*prop.multiProcessorCount);
120  printf(" GPU clock rate: %0.f MHz\n\n", prop.clockRate * 1e-3f);
121 
122  #if CUDART_VERSION >= 5000
123  /* This is supported in CUDA 5.0 (runtime API device properties) */
124  printf(" Memory Clock rate: %.0f Mhz\n", prop.memoryClockRate * 1e-3f);
125  printf(" Memory Bus Width: %d-bit\n", prop.memoryBusWidth);
126 
127  if (prop.l2CacheSize)
128  {
129  printf(" L2 Cache Size: %d bytes\n\n", prop.l2CacheSize);
130  }
131  #endif
132 
133  printf(" Total amount of global memory: %.0f MBytes\n", (float)prop.totalGlobalMem/1048576.0f);
134  printf(" Total amount of constant memory: %lu bytes\n", prop.totalConstMem);
135  printf(" Total amount of shared memory per block: %lu bytes\n", prop.sharedMemPerBlock);
136 
137  /* if prop.major >= 3 set shared memory bank size to 8 byte */
138  if (prop.major >= 3)
139  {
140  cudaDeviceSetSharedMemConfig(cudaSharedMemBankSizeEightByte);
141  printf(" Shared memory bank size %d bytes\n", 8);
142  }
143  else
144  {
145  printf(" Shared memory bank size %d bytes\n", 4);
146  }
147 
148  printf(" Total number of registers available per block: %d\n", prop.regsPerBlock);
149  printf(" Warp size: %d\n\n", prop.warpSize);
150 
151  printf(" Maximum number of threads per multiprocessor: %d\n", prop.maxThreadsPerMultiProcessor);
152  printf(" Maximum number of threads per block: %d\n", prop.maxThreadsPerBlock);
153  printf(" Max dimension size of a thread block (x,y,z): (%d, %d, %d)\n", prop.maxThreadsDim[0],
154 prop.maxThreadsDim[1], prop.maxThreadsDim[2]);
155  printf(" Max dimension size of a grid size (x,y,z): (%d, %d, %d)\n\n", prop.maxGridSize[0],
156 prop.maxGridSize[1], prop.maxGridSize[2]);
157 
158  printf(" Run time limit on kernels: %s\n", prop.kernelExecTimeoutEnabled ? "Yes" : "No");
159  printf(" Integrated GPU sharing Host Memory: %s\n", prop.integrated ? "Yes" : "No");
160  printf(" Support host page-locked memory mapping: %s\n", prop.canMapHostMemory ? "Yes" : "No");
161  printf(" Device has ECC support: %s\n\n", prop.ECCEnabled ? "Enabled" : "Disabled");
162  }
163 }

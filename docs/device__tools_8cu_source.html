35 inline int _ConvertSMVer2Cores(int major, int minor)
40 {
41  /* Defines for GPU Architecture types (using the SM version to determine the # of cores per SM */
42  typedef struct {
43  int SM; /* 0xMm (hexidecimal notation), M = SM Major version, and m = SM minor version */
44  int Cores;
45  } sSMtoCores;
46 
47  sSMtoCores nGpuArchCoresPerSM[] =
48  {
49  { 0x10, 8 }, /* Tesla Generation (SM 1.0) G80 class */
50  { 0x11, 8 }, /* Tesla Generation (SM 1.1) G8x class */
51  { 0x12, 8 }, /* Tesla Generation (SM 1.2) G9x class */
52  { 0x13, 8 }, /* Tesla Generation (SM 1.3) GT200 class */
53  { 0x20, 32 }, /* Fermi Generation (SM 2.0) GF100 class */
54  { 0x21, 48 }, /* Fermi Generation (SM 2.1) GF10x class */
55  { 0x30, 192}, /* Kepler Generation (SM 3.0) GK10x class */
56  { 0x32, 192}, /* Kepler Generation (SM 3.2) GK10x class */
57  { 0x35, 192}, /* Kepler Generation (SM 3.5) GK11x class */
58  { 0x37, 192}, /* Kepler Generation (SM 3.7) GK21x class */
59  { 0x50, 128}, /* Maxwell Generation (SM 5.0) GM10x class */
60  { 0x52, 128},
61  { 0x53, 128},
62  { 0x60, 64},
63  { 0x61, 128},
64  { 0x62, 128},
65  { 0x70, 64},
66  { 0x72, 64},
67  { 0x75, 64},
68  { 0x80, 64},
69  { 0x86, 128},
70  { -1, -1 }
71  };
72 
73  int index = 0;
74 
75  while (nGpuArchCoresPerSM[index].SM != -1) {
76  if (nGpuArchCoresPerSM[index].SM == ((major << 4) + minor)) {
77  return nGpuArchCoresPerSM[index].Cores;
78  }
79  index++;
80  }
81 
82  /* If we don't find the values, we default use the previous one to run properly */
83  printf("MapSMtoCores for SM %d.%d is undefined. Default to use %d Cores/SM\n",
84  major, minor, nGpuArchCoresPerSM[index-1].Cores);
85  return nGpuArchCoresPerSM[index-1].Cores;
86 }

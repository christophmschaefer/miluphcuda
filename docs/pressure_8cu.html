329  pressure_solid = 0.0;
330  } else {

332  if (p.e[i] <= E_iv || eta >= 1.0) {
333  pressure_solid = (a + b / (p.e[i] / (eta * eta * E_0) + 1.0))
334  * p.rho[i] * p.alpha_epspor[i] * p.e[i] + A * mu + B * mu * mu;
335  } else if (p.e[i] >= E_cv && eta >= 0.0) {
336  pressure_solid = a * p.rho[i] * p.alpha_epspor[i] * p.e[i]
337  + (b * p.rho[i] * p.alpha_epspor[i] * p.e[i] / (p.e[i] / (eta * eta * E_0) + 1.0)
338  + A * mu * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_epspor[i]) - 1.0)))
339  * exp(-alpha_till * (pow(rho_0 / (p.rho[i]
340  * p.alpha_epspor[i]) - 1.0, 2)));
341  } else if (p.e[i] > E_iv && p.e[i] < E_cv) {
342  /* for intermediate states:
343  * weighted average of pressures calculated by expanded
344  * and compressed versions of Tillotson (both evaluated at e)
345  */
346  p1 = (a + b / (p.e[i] / (eta * eta * E_0) + 1.0))
347  * p.rho[i] * p.alpha_epspor[i] * p.e[i] + A * mu + B * mu * mu;
348  p2 = a * p.rho[i] * p.alpha_epspor[i] * p.e[i]
349  + (b * p.rho[i] * p.alpha_epspor[i] * p.e[i] / (p.e[i] / (eta * eta * E_0) + 1.0)
350  + A * mu * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_epspor[i]) - 1.0)))
351  * exp(-alpha_till * (pow(rho_0 / (p.rho[i] * p.alpha_epspor[i]) - 1.0, 2)));
352  pressure_solid = ((p.e[i] - E_iv) * p2 + (E_cv - p.e[i]) * p1) / (E_cv - E_iv);
353  } else {
354  printf("Deep trouble in pressure.\n");
355  printf("p[%d].e = %e\n", i, p.e[i]);
356  printf("E_iv: %e, E_cv: %e\n", E_iv, E_cv);
357  pressure_solid = 0.0;
358  }
359  }
360  pressure = pressure_solid / p.alpha_epspor[i]; /* from the P-alpha model which is also used here */
361  p.p[i] = pressure;
362  // printf("Particle: %d \t P: %e \t Alpha: %e \t Rho: %e \t E: %e \t Mu: %e\n", i, pressure, p.alpha_epspor[i], p.rho[i], p.e[i], mu);
363 #endif
364  } else if (EOS_TYPE_REGOLITH == matEOS[matId]) {
365 #if SOLID
366  p.p[i] = 0.0;
367  double I1;
368 #if DIM == 2
369  double shear = matShearmodulus[matId];

373 #else

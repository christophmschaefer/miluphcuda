66 {

68  __shared__ double sharedCourant[NUM_THREADS_LIMITTIMESTEP];

84 #endif

92 #if DIM == 3
93  temp += az*az;
94 #endif
95  sml = p.h[i];
96  if (temp > 0) {
97  temp = sqrt(sml / sqrt(temp));
98  forces = min(forces, temp);
99  }
100  temp = sml / p.cs[i];
101  courant = min(courant, temp);
102  }
103  i = threadIdx.x;
104  sharedForces[i] = forces;
105  sharedCourant[i] = courant;
106  for (j = NUM_THREADS_LIMITTIMESTEP / 2; j > 0; j /= 2) {
107  __syncthreads();
108  if (i < j) {
109  k = i + j;
110  sharedForces[i] = forces = min(forces, sharedForces[k]);
111  sharedCourant[i] = courant = min(courant, sharedCourant[k]);
112  }
113  }
114  // write block result to global memory
115  if (i == 0) {
116  k = blockIdx.x;
117  forcesPerBlock[k] = forces;
118  courantPerBlock[k] = courant;
119  m = gridDim.x - 1;
120  if (m == atomicInc((unsigned int *)&blockCount, m)) {
121  // last block, so combine all block results
122  for (j = 0; j <= m; j++) {
123  forces = min(forces, forcesPerBlock[j]);

134  blockCount = 0;
135  }

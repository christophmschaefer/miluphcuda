50 __global__ void set_stress_tensor(void)
51 {
52  register int i, inc, matId;
53  int d, e;
54  int niters;
55  double sigma[DIM][DIM];
56  double sigmatmp[DIM][DIM];
57  double rotation_matrix[DIM][DIM];
58  double main_stresses[DIM];
59  double max_ev;
60  double damage = 0.0;
61  double stress_damage = 0.0;
62 
63  inc = blockDim.x * gridDim.x;
64  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
65  matId = p_rhs.materialId[i];
66  niters = 0;
67 
68  // here we set the stress tensor sigma from pressure and deviatoric stress S
69  // note, that S was already lowered in plasticity
70 #if FRAGMENTATION
71  damage = pow(p.damage_total[i], DIM);
72  if (damage > 1.0) damage = 1.0;
73  stress_damage = damage;
74 
75  // special handling of granular media with pressure dependent yield strengths
76 # if COLLINS_PRESSURE_DEPENDENT_YIELD_STRENGTH
77  if (!(damage < 1)) {
78  stress_damage = 0.0;
79  }
80 # endif
81 #else
82  damage = 0.0;
83  stress_damage = 0.0;
84 #endif
85 
86 
87 #if DAMAGE_ACTS_ON_PRINCIPAL_STRESSES
88  for (d = 0; d < DIM; d++) {
89  for (e = 0; e < DIM; e++) {
90  sigmatmp[d][e] = 0.0;
91  sigma[d][e] = p.S[stressIndex(i, d, e)];
92  if (d == e) {
93  sigma[d][e] += -p.p[i];
94  }
95  }
96  }
97  // calculate main stresses
98  niters = calculate_all_eigenvalues(sigma, main_stresses, rotation_matrix);
99  for (d = 0; d < DIM; d++) {
100  sigmatmp[d][d] = main_stresses[d];
101  if (sigmatmp[d][d] > 0) {
102  sigmatmp[d][d] *= (1.0 - damage);
103  }
104  }
105  // rotate back the lowered principal stresses
106  multiply_matrix(sigmatmp, rotation_matrix, sigma);
107  transpose_matrix(rotation_matrix);
108  multiply_matrix(rotation_matrix, sigma, sigmatmp);
109 
110  // sigmatmp now holds the stress tensor for particle i with damaged reduced stresses
111  copy_matrix(sigmatmp, sigma);
112 #else
113  for (d = 0; d < DIM; d++) {
114  for (e = 0; e < DIM; e++) {
115  sigma[d][e] = (1.0 - stress_damage) * p.S[stressIndex(i, d, e)];
116  if (d == e) { // the trace
117  if (p.p[i] < 0) {
118  sigma[d][e] += - (1.0 - damage) * p.p[i];
119  } else {
120  sigma[d][e] += -p.p[i];
121  }
122  }
123  }
124  }
125 
126 #endif
127 
128  // remember sigma
129  for (d = 0; d < DIM; d++) {
130  for (e = 0; e < DIM; e++) {
131  p_rhs.sigma[stressIndex(i,d,e)] = sigma[d][e];
132  }
133  }
134 
135  }
136 }
137 #endif

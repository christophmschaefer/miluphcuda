35 __global__ void calculateDensity(int *interactions) {
36  int i;
37  int j;
38  int inc;
39  int ip;
40  int d;
41  double W;
42  double dx[DIM];
43  double dWdx[DIM];
44  double dWdr;
45  double rho;
46  double sml;
47 
48  inc = blockDim.x * gridDim.x;
49  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
50  if (EOS_TYPE_IGNORE == matEOS[p_rhs.materialId[i]] || p_rhs.materialId[i] == EOS_TYPE_IGNORE) {
51  continue;
52  }
53  sml = p.h[i];
54 #if 0
55 #if (DIM == 2)
56  rho = 40.0/(7.0*M_PI*sml*sml) * p.m[i];
57 #elif (DIM == 3)
58  rho = 8.0/(M_PI*sml*sml*sml) * p.m[i];
59 #elif (DIM == 1)
60  rho = 4.0/(3.0*sml) * p.m[i];
61 #endif
62 #endif
63 
64  // self density is m_i W_ii
65  for (d = 0; d < DIM; d++) {
66  dx[d] = 0;
67  }
68  kernel(&W, dWdx, &dWdr, dx, sml);
69  rho = p.m[i] * W;
70 
71 // correction factors for Wendland CX kernels
72  if (kernel == wendlandc2_p) {
73  // these values are for 3D from Dehnen and Aly 2012. aiaiaiaiaiai
74  rho -= rho*0.0294*powf((double) p.noi[i], -0.977);
75  }
76 
77 
78  // sph sum for particle i
79  for (j = 0; j < p.noi[i]; j++) {
80  ip = interactions[i * MAX_NUM_INTERACTIONS + j];
81  if (EOS_TYPE_IGNORE == matEOS[p_rhs.materialId[ip]] || p_rhs.materialId[ip] == EOS_TYPE_IGNORE) {
82  continue;
83  }
84 #if (VARIABLE_SML || INTEGRATE_SML || DEAL_WITH_TOO_MANY_INTERACTIONS)
85  sml = 0.5*(p.h[i] + p.h[ip]);
86 #endif
87 
88  dx[0] = p.x[i] - p.x[ip];
89 #if DIM > 1
90  dx[1] = p.y[i] - p.y[ip];
91 #if DIM > 2
92  dx[2] = p.z[i] - p.z[ip];
93 #endif
94 #endif
95  kernel(&W, dWdx, &dWdr, dx, sml);
96  // contribution of interaction
97  rho += p.m[ip] * W;
98  }
99  // write to global memory
100  p.rho[i] = rho;
101  }
102 }

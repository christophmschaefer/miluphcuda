278 __device__ double bilinear_interpolation_from_linearized(double x, double y, double* table, double* xtab, double* ytab, int ix, int iy, int n_x, int n_y)
279 {
280  double normx, normy, a, b, p;
281 
282  // if (x,y) lies outside table then extrapolate (somewhat linearly) and print a warning
283  if( ix < 0 || iy < 0 )
284  {
285  if( ix < 0 && iy < 0 ) // (x,y) lies in one of the 4 "corners"
286  {
287  if( x < xtab[0] && y < ytab[0] )
288  {
289  normx = (xtab[0]-x) / (xtab[1]-xtab[0]); // (always positive) distance from table end, normalized to x-spacing between 2 outermost table values
290  normy = (ytab[0]-y) / (ytab[1]-ytab[0]); // (always positive) distance from table end, normalized to y-spacing between 2 outermost table values
291  p = table[0] + normx*(table[0]-table[n_y]) + normy*(table[0]-table[1]);
292  }
293  else if( x < xtab[0] && y >= ytab[n_y-1] )
294  {
295  normx = (xtab[0]-x) / (xtab[1]-xtab[0]); // (always positive) distance from table end, normalized to x-spacing between 2 outermost table values
296  normy = (y-ytab[n_y-1]) / (ytab[n_y-1]-ytab[n_y-2]); // (always positive) distance from table end, normalized to y-spacing between 2 outermost table values
297  p = table[n_y-1] + normx*(table[n_y-1]-table[2*n_y-1]) + normy*(table[n_y-1]-table[n_y-2]);
298  }
299  else if( x >= xtab[n_x-1] && y < ytab[0] )
300  {
301  normx = (x-xtab[n_x-1]) / (xtab[n_x-1]-xtab[n_x-2]); // (always positive) distance from table end, normalized to x-spacing between 2 outermost table values
302  normy = (ytab[0]-y) / (ytab[1]-ytab[0]); // (always positive) distance from table end, normalized to y-spacing between 2 outermost table values
303  p = table[(n_x-1)*n_y] + normx*(table[(n_x-1)*n_y]-table[(n_x-2)*n_y]) + normy*(table[(n_x-1)*n_y]-table[(n_x-1)*n_y+1]);
304  }
305  else if( x >= xtab[n_x-1] && y >= ytab[n_y-1] )
306  {
307  normx = (x-xtab[n_x-1]) / (xtab[n_x-1]-xtab[n_x-2]); // (always positive) distance from table end, normalized to x-spacing between 2 outermost table values
308  normy = (y-ytab[n_y-1]) / (ytab[n_y-1]-ytab[n_y-2]); // (always positive) distance from table end, normalized to y-spacing between 2 outermost table values
309  p = table[n_x*n_y-1] + normx*(table[n_x*n_y-1]-table[(n_x-1)*n_y-1]) + normy*(table[n_x*n_y-1]-table[n_x*n_y-2]);
310  }
311  else
312  printf("WARNING: Some odd behavior during extrapolation from ANEOS table encountered for rho = %e and e = %e !\n", x, y);
313  }
314  else if( ix < 0 )
315  {
316  normy = (y-ytab[iy]) / (ytab[iy+1]-ytab[iy]);
317  if( x < xtab[0] )
318  {
319  // linear interpolation in y-direction at xtab[0] and xtab[1]
320  a = table[iy] + normy*(table[iy+1]-table[iy]);

324  p = a + normx*(b-a);
325  }
326  else if( x >= xtab[n_x-1] )
327  {
328  // linear interpolation in y-direction at xtab[n_x-1] and xtab[n_x-2]
329  a = table[(n_x-1)*n_y+iy] + normy*(table[(n_x-1)*n_y+iy+1]-table[(n_x-1)*n_y+iy]);
330  b = table[(n_x-2)*n_y+iy] + normy*(table[(n_x-2)*n_y+iy+1]-table[(n_x-2)*n_y+iy]);
331  // linear extrapolation in x-direction from a and b

337  }

357  p = a + normy*(a-b);

359  else

361  }
362  else
363  printf("WARNING: Some odd behavior during extrapolation from ANEOS table encountered for rho = %e and e = %e !\n", x, y);
364

366  return(p);
367  }
368 
369  // calculate normalized distances of x and y from (lower) table values
370  normx = (x-xtab[ix]) / (xtab[ix+1]-xtab[ix]);

372 
373  // linear interpolation in x-direction at ytab[iy] and ytab[iy+1]

460  // linear extrapolation in x-direction from 'a' and 'b'
461  delta_x_grid = xtab[n_x-1]-xtab[n_x-2];
462  normx = (x-xtab[n_x-1]) / delta_x_grid; // (always positive) distance from table end, normalized to x-spacing between 2 outermost table values
463  *z = a + normx*(a-b);
464  // compute slopes as those of the extrapolation line in x-direction and simply those at the table boundary in the y-direction
465  *dz_dx = (a-b) / delta_x_grid;
466  *dz_dy = (table[(n_x-1)*n_y+iy+1]-table[(n_x-1)*n_y+iy]) / delta_y_grid;
467  }
468  else
469  printf("WARNING: Some odd behavior during extrapolation from ANEOS table encountered for rho = %e and e = %e !\n", x, y);
470  }
471  else if( iy < 0 ) // (x,y) lies either "below" or "above" of the lookup table domain
472  {
473  delta_x_grid = xtab[ix+1]-xtab[ix];
474  normx = (x-xtab[ix]) / delta_x_grid;
475  if( y < ytab[0] ) // (x,y) lies "below" table
476  {
477  // linear interpolation in x-direction at ytab[0] and ytab[1]
478  a = table[ix*n_y] + normx*(table[(ix+1)*n_y]-table[ix*n_y]);
479  b = table[ix*n_y+1] + normx*(table[(ix+1)*n_y+1]-table[ix*n_y+1]);
480  // linear extrapolation in y-direction from 'a' and 'b'
481  delta_y_grid = ytab[1]-ytab[0];
482  normy = (y-ytab[0]) / delta_y_grid; // (always negative) distance from table end, normalized to y-spacing between 2 outermost table values
483  *z = a + normy*(b-a);
484  // compute slopes as those of the extrapolation line in y-direction and simply those at the table boundary in the x-direction
485  *dz_dx = (table[(ix+1)*n_y]-table[ix*n_y]) / delta_x_grid;
486  *dz_dy = (b-a) / delta_y_grid;
487  }
488  else if( y >= ytab[n_y-1] ) // (x,y) lies "above" table
489  {
490  // linear interpolation in x-direction at ytab[n_y-1] and ytab[n_y-2]
491  a = table[(ix+1)*n_y-1] + normx*(table[(ix+2)*n_y-1]-table[(ix+1)*n_y-1]);
492  b = table[(ix+1)*n_y-2] + normx*(table[(ix+2)*n_y-2]-table[(ix+1)*n_y-2]);
493  // linear extrapolation in y-direction from 'a' and 'b'
494  delta_y_grid = ytab[n_y-1]-ytab[n_y-2];
495  normy = (y-ytab[n_y-1]) / delta_y_grid; // (always positive) distance from table end, normalized to y-spacing between 2 outermost table values
496  *z = a + normy*(a-b);
497  // compute slopes as those of the extrapolation line in y-direction and simply those at the table boundary in the x-direction
498  *dz_dx = (table[(ix+2)*n_y-1]-table[(ix+1)*n_y-1]) / delta_x_grid;
499  *dz_dy = (a-b) / delta_y_grid;
500  }
501  else
502  printf("WARNING: Some odd behavior during extrapolation from ANEOS table encountered for rho = %e and e = %e !\n", x, y);
503  }
504  else
505  printf("WARNING: Some odd behavior during extrapolation from ANEOS table encountered for rho = %e and e = %e !\n", x, y);
506 
507  printf("WARNING: Out of ANEOS table for rho = %e and e = %e. Use extrapolated f(rho,e) = %e, df/drho = %e, df/de = %e ...\n", x, y, *z, *dz_dx, *dz_dy);
508  }
509  else // (x,y) lies inside the table
510  {
511  delta_x = x-xtab[ix];
512  delta_y = y-ytab[iy];
513  delta_x_grid = xtab[ix+1]-xtab[ix];
514  delta_y_grid = ytab[iy+1]-ytab[iy];
515  // compute slopes in x-direction at ytab[iy] ('k_a') and ytab[iy+1] ('k_b')
516  k_a = (table[(ix+1)*n_y+iy]-table[ix*n_y+iy]) / delta_x_grid;
517  k_b = (table[(ix+1)*n_y+iy+1]-table[ix*n_y+iy+1]) / delta_x_grid;
518  // compute slope in x-direction as interpolation between 'k_a' and 'k_b'
519  *dz_dx = k_a + delta_y*(k_b-k_a)/delta_y_grid;
520  // linear interpolation in x-direction at ytab[iy] and ytab[iy+1] to obtain z-values 'a' and 'b'
521  a = table[ix*n_y+iy] + delta_x*k_a;
522  b = table[ix*n_y+iy+1] + delta_x*k_b;
523  // compute slope in y-direction between 'a' and 'b'
524  *dz_dy = (b-a) / delta_y_grid;
525  // linear interpolation in y-direction between 'a' and 'b' to obtain target 'z' value
526  *z = a + delta_y*(*dz_dy);
527  }
528 }

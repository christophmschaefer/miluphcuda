321 __global__ void direct_selfgravity()
322 {
323  int i, inc;
324  int j;
325  int d;
326  double a_grav[DIM];
327  double dist;
328  double f;
329  double dx[DIM];
330  double sml;
331 
332  inc = blockDim.x * gridDim.x;
333  // loop over all particles
334  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numRealParticles; i += inc) {
335  for (d = 0; d < DIM; d++) {
336  a_grav[d] = 0.0;
337  }
338  sml = p.h[i];
339  if (p_rhs.materialId[i] == EOS_TYPE_IGNORE || matEOS[p_rhs.materialId[i]] == EOS_TYPE_IGNORE) {
340  continue;
341  }
342 
343  // loop over all other particles
344  for (j = 0; j < numRealParticles; j++) {
345  if (i == j)
346  continue;
347 
348  dist = 0.0;
349  dx[0] = p.x[i] - p.x[j];
350 #if DIM > 1
351  dx[1] = p.y[i] - p.y[j];
352 #if DIM > 2
353  dx[2] = p.z[i] - p.z[j];
354 #endif
355 #endif
356  for (d = 0; d < DIM; d++) {
357  dist += dx[d]*dx[d];
358  }
359  dist = sqrt(dist);
360  f = gravConst * p.m[j]; // / (distance*distance*distance);
361  f /= dist > sml ? (dist*dist*dist) : (sml*sml*sml);
362  for (d = 0; d < DIM; d++) {
363  a_grav[d] -= f*dx[d];
364  }
365  }
366 
367  p.ax[i] += a_grav[0];
368  p.g_ax[i] = a_grav[0];
369 #if DIM > 1
370  p.ay[i] += a_grav[1];
371  p.g_ay[i] = a_grav[1];
372 #if DIM == 3

51 // adds the acceleration due to the point masses
52 __global__ void gravitation_from_point_masses()
53 {
54  int i, inc;
55  int j;
56  int d;
57  double r;
58  double rrr;
59  double dr[DIM];
60  inc = blockDim.x * gridDim.x;
61  // loop for point masses
62  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numPointmasses; i += inc) {
63  for (j = 0; j < numPointmasses; j++) {
64  if (i == j) continue;
65  r = 0.0;
66  dr[0] = pointmass.x[j] - pointmass.x[i];
67 #if DIM > 1
68  dr[1] = pointmass.y[j] - pointmass.y[i];
69 #if DIM > 2
70  dr[2] = pointmass.z[j] - pointmass.z[i];
71 #endif
72 #endif
73  for (d = 0; d < DIM; d++) {
74  r += dr[d]*dr[d];
75  }
76  r = sqrt(r);
77  rrr = r*r*r;
78  pointmass.ax[i] += C_GRAVITY_SI * pointmass.m[j] * dr[0]/(rrr);
79 #if DIM > 1
80  pointmass.ay[i] += C_GRAVITY_SI * pointmass.m[j] * dr[1]/(rrr);
81 #if DIM > 2
82  pointmass.az[i] += C_GRAVITY_SI * pointmass.m[j] * dr[2]/(rrr);
83 #endif
84 #endif
85  }
86  }
87 
88 
89  // loop over all particles
90  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numRealParticles; i += inc) {
91  if (p_rhs.materialId[i] == EOS_TYPE_IGNORE || matEOS[p_rhs.materialId[i]] == EOS_TYPE_IGNORE) {
92  continue;
93  }
94 
95  for (j = 0; j < numPointmasses; j++) {
96  r = 0.0;
97  dr[0] = pointmass.x[j] - p.x[i];
98 #if DIM > 1
99  dr[1] = pointmass.y[j] - p.y[i];
100 #if DIM > 2
101  dr[2] = pointmass.z[j] - p.z[i];
102 #endif
103 #endif
104  for (d = 0; d < DIM; d++) {
105  r += dr[d]*dr[d];
106  }
107  r = sqrt(r);
108  rrr = r*r*r;
109  if (r < pointmass.rmax[j] && r > pointmass.rmin[j]) {
110  p.ax[i] += C_GRAVITY_SI * pointmass.m[j] * dr[0]/(rrr);
111 #if DIM > 1
112  p.ay[i] += C_GRAVITY_SI * pointmass.m[j] * dr[1]/(rrr);
113 #if DIM > 2
114  p.az[i] += C_GRAVITY_SI * pointmass.m[j] * dr[2]/(rrr);
115 #endif
116 #endif
117  } else {
118  p_rhs.materialId[i] = EOS_TYPE_IGNORE;
119  }
120  }
121  }
122 }

124 // compute self gravity using N**2 algorithm
125 __global__ void direct_selfgravity()
126 {
127  int i, inc;
128  int j;
129  int d;
130  double a_grav[DIM];
131  double dist;
132  double f;
133  double dx[DIM];
134  double sml;
135 
136  inc = blockDim.x * gridDim.x;
137  // loop over all particles
138  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numRealParticles; i += inc) {
139  for (d = 0; d < DIM; d++) {
140  a_grav[d] = 0.0;
141  }
142  sml = p.h[i];
143  if (p_rhs.materialId[i] == EOS_TYPE_IGNORE || matEOS[p_rhs.materialId[i]] == EOS_TYPE_IGNORE) {
144  continue;
145  }
146 
147  // loop over all other particles
148  for (j = 0; j < numRealParticles; j++) {
149  if (i == j)
150  continue;
151 
152  dist = 0.0;
153  dx[0] = p.x[i] - p.x[j];
154 #if DIM > 1
155  dx[1] = p.y[i] - p.y[j];
156 #if DIM > 2
157  dx[2] = p.z[i] - p.z[j];
158 #endif
159 #endif
160  for (d = 0; d < DIM; d++) {
161  dist += dx[d]*dx[d];
162  }
163  dist = sqrt(dist);
164  f = C_GRAVITY_SI * p.m[j]; // / (distance*distance*distance);
165  f /= dist > sml ? (dist*dist*dist) : (sml*sml*sml);
166  for (d = 0; d < DIM; d++) {
167  a_grav[d] -= f*dx[d];
168  }
169  }
170 
171  p.ax[i] += a_grav[0];
172  p.g_ax[i] = a_grav[0];
173 #if DIM > 1
174  p.ay[i] += a_grav[1];
175  p.g_ay[i] = a_grav[1];
176 #if DIM == 3
177  p.az[i] += a_grav[2];
178  p.g_az[i] = a_grav[2];
179 #endif
180 #endif
181 
182  }
183 }

185 // compute self gravity using the tree
186 __global__ void selfgravity()
187 {
188  int i, child, nodeIndex, childNumber, depth;
189  double px, ax, dx, f, distance;
190 #if DIM > 1
191  double py, ay, dy;
192 #endif
193  int currentNodeIndex[MAXDEPTH];
194  int currentChildNumber[MAXDEPTH];
195 #if DIM == 3
196  double pz, az, dz;
197 #endif
198  double sml;
199  double thetasq = theta*theta;
200 
201  __shared__ volatile double cellsize[MAXDEPTH];
202  if (0 == threadIdx.x) {
203  cellsize[0] = 4.0 * radius * radius;
204  for (i = 1; i < MAXDEPTH; i++) {
205  cellsize[i] = cellsize[i - 1] * 0.25;
206  }
207  }
208 
209  __syncthreads();
210 
211  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += blockDim.x * gridDim.x) {
212  px = p.x[i];
213 #if DIM > 1
214  py = p.y[i];
215 #if DIM == 3
216  pz = p.z[i];
217 #endif
218 #endif
219  p.g_ax[i] = 0.0;
220 #if DIM > 1
221  p.g_ay[i] = 0.0;
222 #endif
223  sml = p.h[i];
224  ax = 0.0;
225 #if DIM > 1
226  ay = 0.0;
227 #if DIM == 3
228  az = 0.0;
229  p.g_az[i] = 0.0;
230 #endif
231 #endif
232 
233  // start at root
234  depth = 1;
235  currentNodeIndex[depth] = numNodes - 1;
236  currentChildNumber[depth] = 0;
237 
238  do {
239  childNumber = currentChildNumber[depth];
240  nodeIndex = currentNodeIndex[depth];
241 
242  while(childNumber < numChildren) {
243  do {
244  child = childList[childListIndex(nodeIndex, childNumber)];
245  childNumber++;
246  } while(child == EMPTY && childNumber < numChildren);
247  if (child != EMPTY && child != i) { // dont do selfgravity with yourself!
248  dx = p.x[child] - px;
249  distance = dx*dx;
250 #if DIM > 1
251  dy = p.y[child] - py;
252  distance += dy*dy;
253 #endif
254 #if DIM == 3
255  dz = p.z[child] - pz;
256  distance += dz*dz;
257 #endif
258  // if child is leaf or far away
259  //if (child < numParticles || distance * theta > cellsize[depth]) {
260  if (child < numParticles || distance * thetasq > cellsize[depth]) {
261  distance = sqrt(distance);
262  //distance += 1e10;
263  f = C_GRAVITY_SI * p.m[child]; // / (distance*distance*distance);
264  f /= distance > sml ? (distance*distance*distance) : (sml*sml*sml);
265  // f = 0.0;
266  ax += f*dx;
267 #if DIM > 1
268  ay += f*dy;
269 #if DIM == 3
270  az += f*dz;
271 #endif
272 #endif
273  } else {
274  // put child on stack
275  currentChildNumber[depth] = childNumber;
276  currentNodeIndex[depth] = nodeIndex;
277  depth++;
278  if (depth == MAXDEPTH) {
279  printf("\n\nMAXDEPTH reached in selfgravity... this is not good.\n\n");
280  assert(depth < MAXDEPTH);
281  }
282  childNumber = 0;
283  nodeIndex = child;
284  }
285  }
286  }
287  depth--;
288  } while(depth > 0);
289 
290 
291 
292  p.ax[i] += ax;
293  p.g_ax[i] = ax;
294 #if DIM > 1
295  p.ay[i] += ay;
296  p.g_ay[i] = ay;
297 #if DIM == 3
298  p.az[i] += az;
299  p.g_az[i] = az;
300 #endif
301 #endif
302  }
303 }

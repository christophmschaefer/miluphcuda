246 #if DIM == 3

385 __global__ void calculateCentersOfMass()
386 {
387  register int i, k, child, missing;
388  register double m, cm, px;
389 #if DIM > 1
390  register double py;
391 #endif
392 #if DIM == 3
393  register double pz;
394 #endif
395 #if DIM == 3
396  __shared__ volatile int sharedChildList[NUM_THREADS_CALC_CENTER_OF_MASS * 8];
397 #elif DIM == 2
398  __shared__ volatile int sharedChildList[NUM_THREADS_CALC_CENTER_OF_MASS * 4];
399 #elif DIM == 1
400  __shared__ volatile int sharedChildList[NUM_THREADS_CALC_CENTER_OF_MASS * 2];
401 #endif
402 
403  k = maxNodeIndex + (threadIdx.x + blockIdx.x * blockDim.x);
404 
405  missing = 0;
406  while (k < numNodes) {
407  if (missing == 0) {
408  // new cell, so initialize
409  cm = 0.0;
410  px = 0.0;
411 #if DIM > 1
412  py = 0.0;
413 #if DIM == 3
414  pz = 0.0;
415 #endif
416 #endif
417  for (i = 0; i < numChildren; i++) {
418  child = childList[childListIndex(k, i)];
419  if (child != EMPTY) {
420  sharedChildList[missing * NUM_THREADS_CALC_CENTER_OF_MASS + threadIdx.x] = child; // cache missing children
421  m = p.m[child];
422  missing++;
423  if (m >= 0.0) {
424  // child is ready
425  missing--;
426  // add child's contribution
427  cm += m;
428  px += p.x[child] * m;
429 #if DIM > 1
430  py += p.y[child] * m;
431 #if DIM == 3
432  pz += p.z[child] * m;
433 #endif
434 #endif
435  }
436  }
437  }
438  }
439 
440  if (missing != 0) {
441  do {
442  // poll missing child
443  child = sharedChildList[(missing - 1) * NUM_THREADS_CALC_CENTER_OF_MASS + threadIdx.x];
444  m = p.m[child];
445  if (m >= 0.0) {
446  // child is now ready
447  missing--;
448  // add child's contribution
449  cm += m;
450  px += p.x[child] * m;
451 #if DIM > 1
452  py += p.y[child] * m;
453 #if DIM == 3
454  pz += p.z[child] * m;
455 #endif
456 #endif
457  }
458  // repeat until we are done or child is not ready
459  } while ((m >= 0.0) && (missing != 0));
460  }
461 
462  if (missing == 0) {
463  // all children are ready, so store computed information
464  m = 1.0 / cm;
465  p.x[k] = px * m;
466 #if DIM > 1
467  p.y[k] = py * m;
468 #if DIM == 3
469  p.z[k] = pz * m;
470 #endif
471 #endif
472  __threadfence(); // make sure data are visible before setting mass
473  p.m[k] = cm;
474  k += blockDim.x * gridDim.x; // move on to next cell
475  }
476  }

503  noi = p.noi[i];
504  for (j = 0; j < noi; j++) {
505  /* index of interaction partner */
506  indexP = interactions[i * MAX_NUM_INTERACTIONS + j];
507  /* check if i is in interaction list of indexP */
508  found = FALSE;
509  /* loop over all interactions of interaction partner */
510  for (k = 0; k < p.noi[indexP]; k++) {

512  found = TRUE;

514  }
515  }

517  if (!found) {
518  /* remember index, that we want to delete */
519  di[nod++] = j;
520  }
521  }
522  /* remove deleted partners from interaction list */
523  for (k = 0; k < nod; k++) {
524  interactions[i*MAX_NUM_INTERACTIONS+di[k]] = interactions[i*MAX_NUM_INTERACTIONS+noi--];

1038 #if DIM > 1
1039  miny = localMinY;

1041 #if DIM == 3

1043  maxz = localMaxZ;
1044 #endif
1045 #endif
1046  // create root node

1057 #if DIM > 1

1059 #if DIM == 3

1061 #endif
1062 #endif

1065  maxNodeIndex = k;

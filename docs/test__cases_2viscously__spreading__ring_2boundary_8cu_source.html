339 __global__ void insertGhostParticles()
340 {
341  //call with only one block
342  int i, inc, k;
343  volatile int idx;
344 #if SOLID
345  int a, b;
346 #endif
347  double sml;
348  double distance;
349  double normalVel;

351

353  double z;
354 #endif
355  //boundary type: 0 = no slip, 1 = free slip
356  inc = blockDim.x * gridDim.x;
357  for (k = 0; k < numWalls; k++) {
358  __syncthreads();
359  int currentNumParticles = numParticles;
360  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < currentNumParticles; i += inc) {
361  double sml;
362  sml = p.h[i];
363 
364  x = p.x[i];
365  y = p.y[i];
366 #if DIM == 3
367  z = p.z[i];
368 #endif
369 
370 
371  //get distance to wall
372  distance = x*nx[k] + y*ny[k]-d[k];
373 #if DIM == 3
374  distance += z*nz[k];
375 #endif
376 
377 
378  //if distance small enough
379  if (fabs(distance) <= sml/2.0) {
380 
381  //atomic read and increment of numParticles
382  idx = atomicAdd(&numParticles, 1);
383  assert(idx < maxNumParticles);
384 
385 #if 1 // moved to extra function!
386  //mirror particle
387 #if (VARIABLE_SML || INTEGRATE_SML || DEAL_WITH_TOO_MANY_INTERACTIONS)
388  p.h[idx] = sml;
389 #endif
390  p.noi[idx] = p.noi[i];
391  p.cs[idx] = p.cs[i];
392  p.depth[idx] = p.depth[i];
393  p.p[idx] = p.p[i];

399 #endif
400

402  p.x[idx] = x - 2*distance*nx[k];
403  p.y[idx] = y - 2*distance*ny[k];
404 #if DIM == 3
405  p.z[idx] = z - 2*distance*nz[k];
406 #endif
407  /* remember the real particle where the ghost particle
408  originates from */
409  /* for NO_SLIP_BOUNDARY_TYPE, we store i
410  for FREE_SLIP_BOUNDARY_TYPE, we store -i */
411 #if 1
412  if (boundaryType[k] == NO_SLIP_BOUNDARY_TYPE) {
413  p.real_partner[idx] = i+1;
414  } else if (boundaryType[k] == FREE_SLIP_BOUNDARY_TYPE) {
415  p.real_partner[idx] = -i-1;
416  } else {
417  printf("Error: no such boundary type for particle.\n");
418  assert(false);
419  }
420 #endif
421  /* set mass and material type and sml */
422  p.h[idx] = sml;
423  p.materialId[idx] = p.materialId[i];
424 
425  /* all other quantities are set in function setQuantitiesGhostParticles() */
426  if (boundaryType[k] == NO_SLIP_BOUNDARY_TYPE) {
427  //free slip boundary
428  p.vx[idx] = -p.vx[i];
429 #if DIM > 1
430  p.vy[idx] = -p.vy[i];
431 #if DIM == 3
432  p.vz[idx] = -p.vz[i];
433 #endif
434 #endif
435 #if 0
436 #if SOLID
437  for (a = 0; a < DIM; a++) {
438  for (b = 0; b < DIM; b++) {
439  p.S[idx*DIM*DIM+a*DIM+b] = p.S[i*DIM*DIM+a*DIM+b];
440  }
441  }
442 #endif
443 #endif
444 
445  } else if (boundaryType[k] == FREE_SLIP_BOUNDARY_TYPE) {
446  //free slip boundary
447

472 #endif

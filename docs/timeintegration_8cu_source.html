129 #if SOLID
130 __global__ void symmetrizeStress(void) {
131  register int i, j, k, inc;
132  register double val;
133  inc = blockDim.x * gridDim.x;
134  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
135  for (j = 0; j < DIM; j ++) {
136  for (k = 0; k < j; k++) {
137  val = 0.5 * (p.S[stressIndex(i,j,k)] + p.S[stressIndex(i,k,j)]);
138  p.S[stressIndex(i,j,k)] = val;
139  p.S[stressIndex(i,k,j)] = val;
140  }
141  }
142  }
143 }

147 double calculate_angular_momentum(void)
148 {
149  int i;
150  double l_i = 0.0;
151  double Lx = 0.0;
152  double Ly = 0.0;
153  double Lz = 0.0;
154  double L = 0.0;
155 
156 #if DIM > 1
157  for (i = 0; i < numberOfParticles; i++) {
158  l_i = 0;
159 #if DIM > 2
160  l_i = p_host.m[i]*(p_host.y[i]*p_host.vz[i] - p_host.z[i]*p_host.vy[i]);
161  Lx += l_i;
162  l_i = p_host.m[i]*(p_host.z[i]*p_host.vx[i] - p_host.x[i]*p_host.vz[i]);
163  Ly += l_i;
164  l_i = p_host.m[i]*(p_host.x[i]*p_host.vy[i] - p_host.y[i]*p_host.vx[i]);
165  Lz += l_i;
166 #else
167  l_i = p_host.m[i]*(p_host.x[i]*p_host.vy[i] - p_host.y[i]*p_host.vx[i]);
168  Lz += l_i;
169 #endif
170  }
171  L = sqrt(Lx*Lx + Ly*Ly + Lz*Lz);
172 #endif
173 
174  return L;
175 
176 }

178 void initIntegration()
179 {
180 
181 
182  L_ini = calculate_angular_momentum();
183  if (param.verbose) {
184  fprintf(stdout, "Initial angular momentum is: %.17e\n", L_ini);
185  }
186 
187  dt_host = timePerStep;
188  // copy constants to device
189  cudaVerify(cudaMemcpyToSymbol(dt, &dt_host, sizeof(double)));
190  cudaVerify(cudaMemcpyToSymbol(dtmax, &param.maxtimestep, sizeof(double)));
191  cudaVerify(cudaMemcpyToSymbol(theta, &treeTheta, sizeof(double)));
192  cudaVerify(cudaMemcpyToSymbol(numParticles, &numberOfParticles, sizeof(int)));
193  cudaVerify(cudaMemcpyToSymbol(numPointmasses, &numberOfPointmasses, sizeof(int)));
194  cudaVerify(cudaMemcpyToSymbol(maxNumParticles, &maxNumberOfParticles, sizeof(int)));
195  cudaVerify(cudaMemcpyToSymbol(numRealParticles, &numberOfRealParticles, sizeof(int)));
196  cudaVerify(cudaMemcpyToSymbol(numChildren, &numberOfChildren, sizeof(int)));
197  cudaVerify(cudaMemcpyToSymbol(numNodes, &numberOfNodes, sizeof(int)));
198 
199 #if FRAGMENTATION
200  cudaVerify(cudaMemcpyToSymbol(maxNumFlaws, &maxNumFlaws_host, sizeof(int)));
201 #endif
202  // memory for tree
203  cudaVerify(cudaMalloc((void**)&minxPerBlock, sizeof(double)*numberOfMultiprocessors));
204  cudaVerify(cudaMalloc((void**)&maxxPerBlock, sizeof(double)*numberOfMultiprocessors));
205 #if DIM > 1
206  cudaVerify(cudaMalloc((void**)&minyPerBlock, sizeof(double)*numberOfMultiprocessors));
207  cudaVerify(cudaMalloc((void**)&maxyPerBlock, sizeof(double)*numberOfMultiprocessors));
208 #endif
209 #if DIM == 3
210  cudaVerify(cudaMalloc((void**)&minzPerBlock, sizeof(double)*numberOfMultiprocessors));
211  cudaVerify(cudaMalloc((void**)&maxzPerBlock, sizeof(double)*numberOfMultiprocessors));
212 #endif
213 
214  // set the pointer on the gpu to p_device
215  cudaVerify(cudaMemcpyToSymbol(p, &p_device, sizeof(struct Particle)));
216  cudaVerify(cudaMemcpyToSymbol(p_rhs, &p_device, sizeof(struct Particle)));
217 
218  cudaVerify(cudaMemcpyToSymbol(pointmass, &pointmass_device, sizeof(struct Pointmass)));
219  cudaVerify(cudaMemcpyToSymbol(pointmass_rhs, &pointmass_device, sizeof(struct Pointmass)));
220 
221  cudaVerifyKernel((initializeSoundspeed<<<numberOfMultiprocessors*4, NUM_THREADS_512>>>()));
222 }

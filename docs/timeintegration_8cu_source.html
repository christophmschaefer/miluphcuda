120  inc = blockDim.x * gridDim.x;

134 double calculate_angular_momentum(void)
135 {
136  int i;
137  double l_i = 0.0;
138  double Lx = 0.0;
139  double Ly = 0.0;
140  double Lz = 0.0;
141  double L = 0.0;
142 
143 #if DIM > 1
144  for (i = 0; i < numberOfParticles; i++) {
145  l_i = 0;
146 #if DIM > 2
147  l_i = p_host.m[i]*(p_host.y[i]*p_host.vz[i] - p_host.z[i]*p_host.vy[i]);
148  Lx += l_i;
149  l_i = p_host.m[i]*(p_host.z[i]*p_host.vx[i] - p_host.x[i]*p_host.vz[i]);
150  Ly += l_i;
151  l_i = p_host.m[i]*(p_host.x[i]*p_host.vy[i] - p_host.y[i]*p_host.vx[i]);
152  Lz += l_i;
153 #else
154  l_i = p_host.m[i]*(p_host.x[i]*p_host.vy[i] - p_host.y[i]*p_host.vx[i]);
155  Lz += l_i;
156 #endif
157  }
158  L = sqrt(Lx*Lx + Ly*Ly + Lz*Lz);
159 #endif
160 
161  return L;
162 }

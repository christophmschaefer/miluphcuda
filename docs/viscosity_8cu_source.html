34 #if NAVIER_STOKES
35 __global__ void calculate_shear_stress_tensor(int *interactions)
36 {
37 
38  int i, inc;
39  int e, f, g;
40  int j, k;
41  inc = blockDim.x * gridDim.x;
42  //Particle Loop
43  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numRealParticles; i += inc) {
44  double dv[DIM];
45  double dr[DIM];
46  double r;
47  double sml;
48  double dWdr, W;
49  double dWdx[DIM];
50 
51  for (k = 0; k < DIM*DIM; k++) {
52  p.Tshear[i*DIM*DIM+k] = 0.0;
53  }
54 
55  for (k = 0; k < p.noi[i]; k++) {
56  j = interactions[i * MAX_NUM_INTERACTIONS + k];
57 
58  dv[0] = p.vx[i] - p.vx[j];
59 #if DIM > 1
60  dv[1] = p.vy[i] - p.vy[j];
61 #if DIM > 2
62  dv[2] = p.vz[i] - p.vz[j];
63 #endif
64 #endif
65 
66  // relative vector
67  dr[0] = p.x[i] - p.x[j];
68 #if DIM > 1
69  dr[1] = p.y[i] - p.y[j];
70 #if DIM > 2
71  dr[2] = p.z[i] - p.z[j];
72 #endif
73 #endif
74 
75  r = 0;
76  for (e = 0; e < DIM; e++) {
77  r += dr[e]*dr[e];
78  dWdx[e] = 0.0;
79  }
80  W = 0.0;
81  dWdr = 0.0;
82  r = sqrt(r);
83 
84  sml = p.h[i];
85 #if (VARIABLE_SML || INTEGRATE_SML || DEAL_WITH_TOO_MANY_INTERACTIONS)
86  sml = 0.5*(p.h[i] + p.h[j]);
87 #endif
88  // get kernel values for this interaction
89  kernel(&W, dWdx, &dWdr, dr, sml);
90 
91 
92 #if TENSORIAL_CORRECTION
93  for (e = 0; e < DIM; e++) {
94  for (f = 0; f < DIM; f++) {
95  for (g = 0; g < DIM; g++) {
96  p.Tshear[i*DIM*DIM+e*DIM+f] += p.m[j]/p.rho[j] * (p_rhs.tensorialCorrectionMatrix[i*DIM*DIM+f*DIM+g] * (-dv[e]) * dr[g] * dWdr/r + p_rhs.tensorialCorrectionMatrix[i*DIM*DIM+e*DIM+g] * (-dv[f]) * dr[g] * dWdr/r);
97  }
98  // traceless
99  if (e == f) {
100  for (g = 0; g < DIM; g++) {
101  p.Tshear[i*DIM*DIM+e*DIM+f] -= 2./3 * p.m[j]/p.rho[j] * (p_rhs.tensorialCorrectionMatrix[i*DIM*DIM+e*DIM+g] * (-dv[e]) * dr[g] * dWdr/r);
102  }
103 
104  }
105  }
106  }
107 #else
108  double trace = 0;
109  for (e = 0; e < DIM; e++) {
110 # if (SPHEQUATIONS == SPH_VERSION1)
111  trace += p.m[j]/p.rho[i] * (-dv[e])*dWdx[e] ;
112 # elif (SPHEQUATIONS == SPH_VERSION2)
113  trace += p.m[j]/p.rho[j] * (-dv[e])*dWdx[e] ;
114 #endif
115  }
116 
117  for (e = 0; e < DIM; e++) {
118  for (f = 0; f < DIM; f++) {
119 # if (SPHEQUATIONS == SPH_VERSION1)
120  p.Tshear[i*DIM*DIM+e*DIM+f] += p.m[j]/p.rho[i] * (-dv[e]*dWdx[f] - dv[f]*dWdx[e]);
121 # elif (SPHEQUATIONS == SPH_VERSION2)
122  p.Tshear[i*DIM*DIM+e*DIM+f] += p.m[j]/p.rho[j] * (-dv[e]*dWdx[f] - dv[f]*dWdx[e]);
123 #endif
124  // traceless
125  if (e == f) {
126 # if (SPHEQUATIONS == SPH_VERSION1)
127  p.Tshear[i*DIM*DIM+e*DIM+f] -= 2./3 * trace;
128 # elif (SPHEQUATIONS == SPH_VERSION2)
129  p.Tshear[i*DIM*DIM+e*DIM+f] -= 2./3 * trace;
130 #endif
131  }
132  }
133  }
134 #endif
135  }
136  }
137 }
138 #endif

141 /*normally known as beta viscosity, here we use beta due to naming concerns of other variables in this code and to avoid confusion */
142 #if INVISCID_SPH
143 __global__ void betaviscosity(int *interactions)
144 {
145 
146  register int d, i, j, k, m, inc, numInteractions;
147  inc = blockDim.x * gridDim.x;
148 
149  double dv[DIM], dx[DIM];
150  double vi[DIM], vj[DIM];
151  double mj, rhoi;
152  double W, dWdx[DIM], dWdr;
153  double divv[DIM], divV, divvdt;
154  double tau, sml, csound;
155  double Ai, Ri, Xii;
156  double curlj[DIM][DIM] = {0};
157  double transpose[DIM][DIM] = {0};
158  double limiterMatrix[DIM][DIM] = {0};
159  double traceLimiter = 0;
160  double beta_loc;
161  double beta_max = 4;
162 
163  //Particle Loop
164  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numRealParticles; i += inc) {
165  numInteractions = p.noi[i];
166  sml = p.h[i];
167  csound = p.cs[i];
168  rhoi = p.rho[i];
169  W = 0.0;
170  dWdr = 0.0;
171  Ai = 0;
172  Ri = 0;
173  Xii = 0;
174 
175  //Interaction Partner Loop
176  for(k = 0; k < numInteractions; k++) {
177  j = interactions[i * MAX_NUM_INTERACTIONS + k];
178  dx[0] = p.x[i] - p.x[j];
179  dv[0] = p.vx[i] - p.vx[j];
180 #if DIM > 1
181  dx[1] = p.y[i] - p.y[j];
182  dv[1] = p.vy[i] - p.vy[j];
183 #if DIM > 2
184  dx[2] = p.z[i] - p.z[j];
185  dv[2] = p.vz[i] - p.vz[j];
186 #endif
187 #endif
188  vi[0] = p.vx[i];
189  vj[0] = p.vx[j];
190 #if DIM > 1
191  vi[1] = p.vy[i];
192  vj[1] = p.vy[j];
193 #if DIM > 2
194  vi[2] = p.vz[i];
195  vj[2] = p.vz[j];
196 #endif
197 #endif
198 
199  kernel(&W, dWdx, &dWdr, dx, sml);
200  mj = p.m[j];
201 
202  /* divv */
203  for (d = 0; d < DIM; d++) {
204  divv[d] = mj/rhoi * (vj[d] - vi[d]) * dWdx[d];
205  /* Limiter Matrix */
206  for (m = 0; m < DIM; m++) {
207  curlj[d][m] = 0;
208  }
209  }
210 
211 
212  for (d = 0; d < DIM; d++) {
213  curlj[d][d] = -1.0/DIM * divv[d];
214  for (m = 0; m < DIM; m++) {
215  curlj[d][m] += (mj/rhoi) *(dv[d]*dWdx[d]/(dx[m]*dWdx[m]) + dv[m]*dWdx[m]/(dx[d]*dWdx[d]))/2;
216  transpose[m][d] = curlj[d][m];
217  }
218  }
219 
220  multiply(curlj, transpose, limiterMatrix);
221  for (d = 0; d < DIM; d++) {
222  traceLimiter += limiterMatrix[d][d];
223  Ri += 1.0/rhoi * sign(dv[d]) * mj * dWdx[d];
224  }
225  } /* Ending Interaction Partner Loop */
226 
227 
228 
229 
230  /* Calculating d/dt(divv) */
231  divV = p_rhs.divv[i];
232  divvdt = divV - p.divv_old[i];
233 
234 
235  /* decay time */
236  tau = sml / (2.0 * 0.05 * csound);
237 
238  /* Limiter for strong shear forces */
239  Xii = pow(fabs(2 * pow((1.0 - Ri),4) * divV),2) / (pow(fabs(2 * pow((1.0 - Ri),4) * divV),2) + pow(1.0+fabs(traceLimiter),2));
240  //fprintf(stdout, "Ri: %e\n", Ri);
241  //fprintf(stdout, "vvnablaW: %e\n", vvnablaW);
242  //fprintf(stdout, "traceLimiter: %e\n", traceLimiter);
243  //fprintf(stdout, "Xii: %e\n", Xii);
244 
245  /* Calc shock indicator */
246  if ((-1.0*divvdt) > 0) {
247  Ai = Xii * -1.0 * divvdt;
248  }
249  else {
250  Ai = 0;
251  }
252  //fprintf(stdout, "p[%d].divvdt: %e\n", i, p[i].divvdt);
253  beta_loc = beta_max * pow(sml, 2) * Ai / (pow(csound, 2) + pow(sml, 2) * Ai);
254  //fprintf(stdout, "p[%d].cs: %e\n", i, p[i].cs);
255  //fprintf(stdout, "p[%d].sml: %e\n", i, p[i].sml);
256  //fprintf(stdout, "Ai: %e\n", Ai);
257  //fprintf(stdout, "beta_max: %e\n", beta_max);
258  //fprintf(stdout, "beta_loc: %e\n", beta_loc);
259 
260  if (beta_loc > p.beta_old[i]) {
261  p.beta[i] = beta_loc;
262  p.dbetadt[i] = 0;
263  } else
264  {
265  p.beta[i] = p.beta_old[i];
266  p.dbetadt[i] = (p.beta_old[i] - beta_loc) / tau;
267  }
268 
269  /* getting ready for the next timestep by giving the "new" old values the current ones */
270  p.beta_old[i] = p.beta[i];
271  p.divv_old[i] = divV;
272  } // Ending Particle Loop
273 }
274 #endif

39 /* the runge-kutta 4nd order integrator with fixed timestep */
40 void rk4_nbodies()
41 {
42  int rkstep;
43 
44  // alloc mem for multiple rhs and copy immutables
45  int allocate_immutables = 0;
46  for (rkstep = 0; rkstep < 4; rkstep++) {
47  copy_pointmass_immutables_device_to_device(&rk4_pointmass_device[rkstep], &pointmass_device);
48  }
49 
50  copy_pointmass_variables_device_to_device(&rk4_pointmass_device[RKFIRST], &pointmass_device);
51  copy_pointmass_variables_device_to_device(&rk4_pointmass_device[RKSTART], &pointmass_device);
52  cudaVerify(cudaDeviceSynchronize());
53 
54  // calculate first right hand side with rk[RKFIRST]_device
55  cudaVerify(cudaMemcpyToSymbol(pointmass, &rk4_pointmass_device[RKFIRST], sizeof(struct Pointmass)));
56  cudaVerifyKernel((rhs_pointmass<<<numberOfMultiprocessors, NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
57  cudaVerify(cudaDeviceSynchronize());
58 
59  // remember values of first step
60  copy_pointmass_variables_device_to_device(&rk4_pointmass_device[RKSTART], &rk4_pointmass_device[RKFIRST]);
61  copy_pointmass_derivatives_device_to_device(&rk4_pointmass_device[RKSTART], &rk4_pointmass_device[RKFIRST]);
62  cudaVerify(cudaDeviceSynchronize());
63  // set rk[RKFIRST] variables
64  cudaVerifyKernel((rhs_pointmass<<<numberOfMultiprocessors,NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
65  cudaVerifyKernel((rk4_integrateFirstStep<<<numberOfMultiprocessors, NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
66  cudaVerify(cudaDeviceSynchronize());
67 
68  // get derivatives for second step
69  cudaVerify(cudaMemcpyToSymbol(pointmass, &rk4_pointmass_device[RKFIRST], sizeof(struct Pointmass)));
70  cudaVerifyKernel((rhs_pointmass<<<numberOfMultiprocessors,NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
71  cudaVerify(cudaDeviceSynchronize());
72  // rk4_integrate second step
73  cudaVerifyKernel((rk4_integrateSecondStep<<<numberOfMultiprocessors, NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
74  cudaVerify(cudaDeviceSynchronize());
75 
76  // get derivatives for third step
77  cudaVerify(cudaMemcpyToSymbol(pointmass, &rk4_pointmass_device[RKSECOND], sizeof(struct Pointmass)));
78  cudaVerifyKernel((rhs_pointmass<<<numberOfMultiprocessors,NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
79  // rk4_integrate third step
80  cudaVerifyKernel((rk4_integrateThirdStep<<<numberOfMultiprocessors, NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
81 
82  // get derivatives for the fourth (and last) step
83  // this happens at t = t0 + h
84  cudaVerify(cudaMemcpyToSymbol(pointmass, &rk4_pointmass_device[RKTHIRD], sizeof(struct Pointmass)));
85  cudaVerifyKernel((rhs_pointmass<<<numberOfMultiprocessors,NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
86  cudaVerify(cudaMemcpyToSymbol(pointmass, &pointmass_device, sizeof(struct Pointmass)));
87  // rk4_integrate fourth step
88  cudaVerifyKernel((rk4_integrateFourthStep<<<numberOfMultiprocessors, NUM_THREADS_RK4_INTEGRATE_STEP>>>()));
89 }

91 // acceleration due to the point masses
92 __global__ void rhs_pointmass()
93 {
94  int i, inc;
95  int j;
96  int d;
97  double r;
98  double rrr;
99  double dr[DIM];
100  inc = blockDim.x * gridDim.x;
101  // loop for point masses
102  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numPointmasses; i += inc) {
103  pointmass.ax[i] = 0.0;
104 #if DIM > 1
105  pointmass.ay[i] = 0.0;
106 #if DIM > 2
107  pointmass.az[i] = 0.0;
108 #endif
109 #endif
110  for (j = 0; j < numPointmasses; j++) {
111  if (i == j) continue;
112  r = 0.0;
113  dr[0] = pointmass.x[j] - pointmass.x[i];
114 #if DIM > 1
115  dr[1] = pointmass.y[j] - pointmass.y[i];
116 #if DIM > 2
117  dr[2] = pointmass.z[j] - pointmass.z[i];
118 #endif
119 #endif
120  for (d = 0; d < DIM; d++) {
121  r += dr[d]*dr[d];
122  }
123  r = sqrt(r);
124  rrr = r*r*r;
125  pointmass.ax[i] += C_GRAVITY * pointmass.m[j] * dr[0]/(rrr);
126 #if DIM > 1
127  pointmass.ay[i] += C_GRAVITY * pointmass.m[j] * dr[1]/(rrr);
128 #if DIM > 2
129  pointmass.az[i] += C_GRAVITY * pointmass.m[j] * dr[2]/(rrr);
130 #endif
131 #endif
132  }
133  if (pointmass_rhs.feels_particles[i]) {
134  pointmass.ax[i] += pointmass_rhs.feedback_ax[i];
135 #if DIM > 1
136  pointmass.ay[i] += pointmass_rhs.feedback_ay[i];
137 #if DIM > 2
138  pointmass.az[i] += pointmass_rhs.feedback_az[i];
139 #endif
140 #endif
141  }
142  }
143 }

145 __global__ void rk4_integrateFirstStep()
146 {
147  int i;
148  // loop for the point masses
149  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numPointmasses; i+= blockDim.x * gridDim.x) {
150  rk4_pointmass[RKFIRST].x[i] = rk4_pointmass[RKSTART].x[i] + dt * b21 * rk4_pointmass[RKFIRST].vx[i];
151 #if DIM > 1
152  rk4_pointmass[RKFIRST].y[i] = rk4_pointmass[RKSTART].y[i] + dt * b21 * rk4_pointmass[RKFIRST].vy[i];
153 #endif
154 #if DIM == 3
155  rk4_pointmass[RKFIRST].z[i] = rk4_pointmass[RKSTART].z[i] + dt * b21 * rk4_pointmass[RKFIRST].vz[i];
156 #endif
157 
158  rk4_pointmass[RKFIRST].vx[i] = rk4_pointmass[RKSTART].vx[i] + dt * b21 * rk4_pointmass[RKFIRST].ax[i];
159 #if DIM > 1
160  rk4_pointmass[RKFIRST].vy[i] = rk4_pointmass[RKSTART].vy[i] + dt * b21 * rk4_pointmass[RKFIRST].ay[i];
161 #endif
162 #if DIM == 3
163  rk4_pointmass[RKFIRST].vz[i] = rk4_pointmass[RKSTART].vz[i] + dt * b21 * rk4_pointmass[RKFIRST].az[i];
164 #endif
165  }
166 }

214 __global__ void rk4_integrateFourthStep()
215 {
216  int i;
217  int d;
218  // loop pointmasses
219  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numPointmasses; i+= blockDim.x * gridDim.x) {
220  pointmass.vx[i] = rk4_pointmass[RKSTART].vx[i] + dt/6.0 * (c1 * rk4_pointmass[RKSTART].ax[i] + b32 * rk4_pointmass[RKFIRST].ax[i] + b32 * rk4_pointmass[RKSECOND].ax[i] + c1 * rk4_pointmass[RKTHIRD].ax[i]);
221  pointmass.ax[i] = 1./6.0 *(c1 * rk4_pointmass[RKSTART].ax[i] + b32 * rk4_pointmass[RKFIRST].ax[i] + b32 * rk4_pointmass[RKSECOND].ax[i] + c1 * rk4_pointmass[RKTHIRD].ax[i]);
222 #if DIM > 1
223  pointmass.vy[i] = rk4_pointmass[RKSTART].vy[i] + dt/6.0 * (c1 * rk4_pointmass[RKSTART].ay[i] + b32 * rk4_pointmass[RKFIRST].ay[i] + b32 * rk4_pointmass[RKSECOND].ay[i] + c1 * rk4_pointmass[RKTHIRD].ay[i]);
224  pointmass.ay[i] = 1./6.0 *(c1 * rk4_pointmass[RKSTART].ay[i] + b32 * rk4_pointmass[RKFIRST].ay[i] + b32 * rk4_pointmass[RKSECOND].ay[i] + c1 * rk4_pointmass[RKTHIRD].ay[i]);
225 #endif
226 #if DIM > 2
227  pointmass.vz[i] = rk4_pointmass[RKSTART].vz[i] + dt/6.0 * (c1 * rk4_pointmass[RKSTART].az[i] + b32 * rk4_pointmass[RKFIRST].az[i] + b32 * rk4_pointmass[RKSECOND].az[i] + c1 * rk4_pointmass[RKTHIRD].az[i]);
228  pointmass.az[i] = 1./6.0 *(c1 * rk4_pointmass[RKSTART].az[i] + b32 * rk4_pointmass[RKFIRST].az[i] + b32 * rk4_pointmass[RKSECOND].az[i] + c1 * rk4_pointmass[RKFIRST].az[i]);
229 #endif
230 
231  pointmass.x[i] = rk4_pointmass[RKSTART].x[i] + dt/6.0 * (c1 * rk4_pointmass[RKSTART].vx[i] + b32 * rk4_pointmass[RKFIRST].vx[i] + b32 * rk4_pointmass[RKSECOND].vx[i] + c1 * rk4_pointmass[RKTHIRD].vx[i]);
232 #if DIM > 1
233  pointmass.y[i] = rk4_pointmass[RKSTART].y[i] + dt/6.0 * (c1 * rk4_pointmass[RKSTART].vy[i] + b32 * rk4_pointmass[RKFIRST].vy[i] + b32 * rk4_pointmass[RKSECOND].vy[i] + c1 * rk4_pointmass[RKTHIRD].vy[i]);

90 __global__ void rhs_pointmass()
91 {
92  int i, inc;
93  int j;
94  int d;
95  double r;
96  double rrr;
97  double dr[DIM];
98  inc = blockDim.x * gridDim.x;
99  // loop for point masses
100  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numPointmasses; i += inc) {
101  pointmass.ax[i] = 0.0;
102 #if DIM > 1
103  pointmass.ay[i] = 0.0;
104 #if DIM > 2
105  pointmass.az[i] = 0.0;
106 #endif
107 #endif
108  for (j = 0; j < numPointmasses; j++) {
109  if (i == j) continue;
110  r = 0.0;
111  dr[0] = pointmass.x[j] - pointmass.x[i];
112 #if DIM > 1
113  dr[1] = pointmass.y[j] - pointmass.y[i];
114 #if DIM > 2
115  dr[2] = pointmass.z[j] - pointmass.z[i];
116 #endif
117 #endif
118  for (d = 0; d < DIM; d++) {
119  r += dr[d]*dr[d];
120  }
121  r = sqrt(r);
122  rrr = r*r*r;
123  pointmass.ax[i] += gravConst * pointmass.m[j] * dr[0]/(rrr);
124 #if DIM > 1
125  pointmass.ay[i] += gravConst * pointmass.m[j] * dr[1]/(rrr);
126 #if DIM > 2
127  pointmass.az[i] += gravConst * pointmass.m[j] * dr[2]/(rrr);
128 #endif
129 #endif
130  }
131  if (pointmass_rhs.feels_particles[i]) {
132  pointmass.ax[i] += pointmass_rhs.feedback_ax[i];
133 #if DIM > 1
134  pointmass.ay[i] += pointmass_rhs.feedback_ay[i];
135 #if DIM > 2
136  pointmass.az[i] += pointmass_rhs.feedback_az[i];
137 #endif
138 #endif
139  }
140  }

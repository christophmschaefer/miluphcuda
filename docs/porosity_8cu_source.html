65  register int i, inc, matId;
66  inc = blockDim.x * gridDim.x;
67  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
68  matId = p_rhs.materialId[i];
69  if (matEOS[matId] == EOS_TYPE_SIRONO) {
70  double alpha = matporsirono_alpha[matId];
71  double pm = matporsirono_pm[matId];
72  double phimax = matporsirono_phimax[matId];
73  double phi0 = matporsirono_phi0[matId];
74  double delta = matporsirono_delta[matId];
75  double rho_s = matporsirono_rho_s[matId];
76  double phi = p.rho[i] / rho_s;
77  /* Using omni-sided_compression curve for compressive strength */
78  if (phi <= 0.125)
79  p.compressive_strength[i] = alpha * 31.45166;
80  if ((phi > 0.125) && (phi < 0.58))
81  p.compressive_strength[i] = alpha * pm * pow(((phimax - phi0) / (phimax - phi) - 1.0), delta * 2.302585);
82  if (phi >= 0.58)
83  p.compressive_strength[i] = MAXFLOAT;
84  } else {

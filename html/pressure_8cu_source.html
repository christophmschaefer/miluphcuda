30 __global__ void calculatePressure() {
31  register int i, inc, matId;
32  register double eta, e, rho, mu, p1, p2;
33  int i_rho, i_e;
34  double pressure;
35  inc = blockDim.x * gridDim.x;
36  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < numParticles; i += inc) {
37  pressure = 0.0;
38  matId = p_rhs.materialId[i];
39  if (EOS_TYPE_IGNORE == matEOS[matId] || matId == EOS_TYPE_IGNORE) {
40  continue;
41  }
42  if (EOS_TYPE_POLYTROPIC_GAS == matEOS[matId]) {
43  p.p[i] = matPolytropicK[matId] * pow(p.rho[i], matPolytropicGamma[matId]);
44  } else if (EOS_TYPE_IDEAL_GAS == matEOS[matId]) {
45  p.p[i] = (matPolytropicGamma[matId] - 1) * p.rho[i] * p.e[i];
46  } else if (EOS_TYPE_LOCALLY_ISOTHERMAL_GAS == matEOS[matId]) {
47  p.p[i] = p.cs[i]*p.cs[i] * p.rho[i];
48  } else if (EOS_TYPE_ISOTHERMAL_GAS == matEOS[matId]) {
49  /* this is pure molecular hydrogen at 10 K */
50  p.p[i] = 41255.407 * p.rho[i];
51  } else if (EOS_TYPE_MURNAGHAN == matEOS[matId] || EOS_TYPE_VISCOUS_REGOLITH == matEOS[matId]) {
52  eta = p.rho[i] / matRho0[matId];
53  if (eta < matRhoLimit[matId]) {
54  p.p[i] = 0.0;
55  } else {
56  p.p[i] = (matBulkmodulus[matId]/matN[matId])*(pow(eta, matN[matId]) - 1.0);
57  }
58  } else if (EOS_TYPE_TILLOTSON == matEOS[matId]) {
59  rho = p.rho[i];
60  e = p.e[i];
61  eta = rho / matTillRho0[matId];
62  mu = eta - 1.0;
63  if (eta < matRhoLimit[matId] && e < matTillEcv[matId]) {
64  p.p[i] = 0.0;
65  } else {
66  if (e <= matTillEiv[matId] || eta >= 1.0) {
67  p.p[i] = (matTilla[matId] + matTillb[matId]/(e/(eta*eta*matTillE0[matId])+1.0))
68  * rho * e + matTillA[matId]*mu + matTillB[matId]*mu*mu;
69  } else if (e >= matTillEcv[matId] && eta >= 0.0) {
70  p.p[i] = matTilla[matId]*rho*e + (matTillb[matId]*rho*e/(e/(eta*eta*matTillE0[matId])+1.0)
71  + matTillA[matId] * mu * exp(-matTillBeta[matId]*(matTillRho0[matId]/rho - 1.0)))
72  * exp(-matTillAlpha[matId] * (pow(matTillRho0[matId]/rho-1.0, 2)));
73  } else if (e > matTillEiv[matId] && e < matTillEcv[matId]) {
74  // for intermediate states:
75  // weighted average of pressures calculated by expanded
76  // and compressed versions of Tillotson (both evaluated at e)
77  p1 = (matTilla[matId]+matTillb[matId]/(e/(eta*eta*matTillE0[matId])+1.0)) * rho*e
78  + matTillA[matId]*mu + matTillB[matId]*mu*mu;
79  p2 = matTilla[matId]*rho*e + (matTillb[matId]*rho*e/(e/(eta*eta*matTillE0[matId])+1.0)
80  + matTillA[matId] * mu * exp(-matTillBeta[matId]*(matTillRho0[matId]/rho -1.0)))
81  * exp(-matTillAlpha[matId] * (pow(matTillRho0[matId]/rho-1.0, 2)));
82  p.p[i] = ( p1*(matTillEcv[matId]-e) + p2*(e-matTillEiv[matId]) ) / (matTillEcv[matId]-matTillEiv[matId]);
83  } else {
84  printf("\n\nDeep trouble in pressure.\nenergy[%d] = %e\nE_iv = %e, E_cv = %e\n\n", i, e, matTillEiv[matId], matTillEcv[matId]);
85  p.p[i] = 0.0;
86  }
87  }
88  } else if (EOS_TYPE_ANEOS == matEOS[matId]) {
89  /* find array-indices just below the actual values of rho and e */
90  i_rho = array_index(p.rho[i], aneos_rho_c+aneos_rho_id_c[matId], aneos_n_rho_c[matId]);
91  i_e = array_index(p.e[i], aneos_e_c+aneos_e_id_c[matId], aneos_n_e_c[matId]);
92  /* interpolate (bi)linearly to obtain the pressure and other quantities of interest*/
93  p.p[i] = bilinear_interpolation_from_linearized(p.rho[i], p.e[i], aneos_p_c+aneos_matrix_id_c[matId], aneos_rho_c+aneos_rho_id_c[matId], aneos_e_c+aneos_e_id_c[matId], i_rho, i_e, aneos_n_rho_c[matId], aneos_n_e_c[matId]);
94 #if SIRONO_POROSITY
95  } else if (matEOS[matId] == EOS_TYPE_SIRONO) {
96  double K_0 = matporsirono_K_0[matId];
97  double rho_0 = matporsirono_rho_0[matId];
98  double gamma_K = matporsirono_gamma_K[matId];
99  pressure = p.K[i] * (p.rho[i] / p.rho_0prime[i] - 1.0);
100  p.flag_rho_0prime[i] = -1;
101  if (pressure >= 0.0) {
102  if (p.rho[i] >= p.rho_c_plus[i]) {
103  if (pressure >= p.compressive_strength[i]) {
104  pressure = p.compressive_strength[i];
105  p.flag_plastic[i] = 1;
106  p.rho_c_plus[i] = p.rho[i];
107  p.flag_rho_0prime[i] = 1;
108  } else {
109  p.flag_plastic[i] = -1;
110  p.flag_rho_0prime[i] = -1;
111  }
112  } else {
113  if (pressure >= p.compressive_strength[i]) {
114  pressure = p.compressive_strength[i];
115  }
116  if (p.flag_plastic[i] == 1) {
117  p.flag_rho_0prime[i] = 1;
118  p.flag_plastic[i] = -1;
119  } else {
120  p.flag_rho_0prime[i] = -1;
121  p.flag_plastic[i] = -1;
122  }
123  }
124  } else {
125  if (p.rho[i] <= p.rho_c_minus[i]) {
126  if (pressure <= p.tensile_strength[i]) {
127  pressure = p.tensile_strength[i];
128  p.flag_plastic[i] = 1;
129  p.rho_c_minus[i] = p.rho[i];
130  p.flag_rho_0prime[i] = 1;
131  } else {
132  p.flag_plastic[i] = -1;
133  p.flag_rho_0prime[i] = -1;
134  }
135  } else {
136  if (pressure <= p.tensile_strength[i]) {
137  pressure = p.tensile_strength[i];
138  }
139  if (p.flag_plastic[i] == 1) {
140  p.flag_rho_0prime[i] = 1;
141  p.flag_plastic[i] = -1;
142  } else {
143  p.flag_plastic[i] = -1;
144  p.flag_rho_0prime[i] = -1;
145  }
146  }
147  }
148  /* determine new rho_0prime and K if flag is set */
149  if (p.flag_rho_0prime[i] == 1) {
150  p.flag_rho_0prime[i] = -1;
151  p.flag_plastic[i] = -1;
152  p.K[i] = K_0 * pow((p.rho[i] / rho_0), gamma_K);
153  if (p.K[i] < 2000.0) {
154  p.K[i] = 2000.0;
155  printf("p.K[%d] is small and set to 2000.0", i);
156  }
157  p.rho_0prime[i] = p.rho[i] / (1.0 + (pressure / p.K[i]));
158  }
159  p.p[i] = pressure;
160 #endif
161 #if PALPHA_POROSITY
162  } else if (matEOS[matId] == EOS_TYPE_JUTZI || matEOS[matId] == EOS_TYPE_JUTZI_MURNAGHAN) {
163  double pressure_solid = 0.0;
164  double p_e = matporjutzi_p_elastic[matId]; /* pressure at which the material switches from elastic to plastic */
165  double p_t = matporjutzi_p_transition[matId]; /* pressure indicating a transition */
166  double p_s = matporjutzi_p_compacted[matId]; /* pressure at which all pores are compacted (alpha = 1) */
167  double alpha_0 = matporjutzi_alpha_0[matId]; /* distention at which the material switches from elastic to plastic */
168  double alpha_t = matporjutzi_alpha_t[matId]; /* distention indicating a transition */
169  double n1 = matporjutzi_n1[matId]; /* individual slope */
170  double n2 = matporjutzi_n2[matId]; /* individual slope */
171  double alpha_e = matporjutzi_alpha_e[matId]; /* simplified otherwise alpha_e = alpha(p_e) */
172 // p.alpha_jutzi_old[i] = p.alpha_jutzi[i]; /* saving the unchanged alpha value */
173  int flag_alpha_quad; /* if this flag is set -> alpha gets calculated by a quadradic equation and not via the crush curve */
174  double dp; /* pressure change for the calculation of dalphadp */
175  double rho_0 = matTillRho0[matId]; /* parameters for Tillotson EOS -> calc pressure solid */
176  double eta = p.rho[i] * p.alpha_jutzi[i] / rho_0;
177  int crushcurve_style = matcrushcurve_style[matId]; /* int crushcurve_style from mat.cfg -> 0 is the quadratic crush curve, 1 is the real/steep crush curve by jutzi */
178  if (matEOS[matId] == EOS_TYPE_JUTZI) {
179  double alpha_till = matTillAlpha[matId];
180  double beta_till = matTillBeta[matId];
181  double a = matTilla[matId];
182  double b = matTillb[matId];
183  double A = matTillA[matId];
184  double B = matTillB[matId];
185  double E_0 = matTillE0[matId];
186  double E_iv = matTillEiv[matId];
187  double E_cv = matTillEcv[matId];
188  p.delpdele[i] = 0.0;
189  p.delpdelrho[i] = 0.0;
190  /* calculate the pressure of the solid material and also
191  * calculate the derivative del p / del e and the derivative del p / del rho */
192  if (eta < matRhoLimit[matId] && p.e[i] < E_cv) {
193  pressure_solid = 0.0;
194  } else {
195  mu = eta - 1.0;
196  if (p.e[i] < E_iv || eta >= 1.0) {
197  pressure_solid = (a + b / (p.e[i] / (eta * eta * E_0) + 1.0))
198  * p.rho[i] * p.alpha_jutzi[i] * p.e[i] + A * mu + B * mu * mu;
199  p.delpdele[i] = a * p.rho[i] * p.alpha_jutzi[i] + p.rho[i] * p.alpha_jutzi[i]
200  * b / (pow(p.e[i] / (E_0 * eta * eta) + 1.0, 2));
201  p.delpdelrho[i] = a * p.e[i] + p.e[i] * b * (1.0 + 3.0 * p.e[i] / (E_0 * eta * eta))
202  / (pow(p.e[i] / (E_0 * eta * eta) + 1.0, 2))
203  + A / rho_0 + 2.0 * B / rho_0 * (eta - 1.0);
204  } else if (p.e[i] > E_cv && eta < 1.0) {
205  pressure_solid = a * p.rho[i] * p.alpha_jutzi[i] * p.e[i]
206  + (b * p.rho[i] * p.alpha_jutzi[i] * p.e[i] / (p.e[i] / (eta * eta * E_0) + 1.0)
207  + A * mu * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0)))
208  * exp(-alpha_till * (pow(rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0, 2)));
209  p.delpdele[i] = a * p.rho[i] * p.alpha_jutzi[i] + p.rho[i] * p.alpha_jutzi[i] * b / (pow(p.e[i]/(E_0 * eta * eta) + 1.0, 2))
210  * exp(-alpha_till * (pow(rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0, 2)));
211  p.delpdelrho[i] = a * p.e[i] + exp(-alpha_till * (pow(rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0, 2)))
212  * (2.0 * alpha_till * rho_0 / (p.rho[i] * p.rho[i] * p.alpha_jutzi[i]
213  * p.alpha_jutzi[i]) * (rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0)
214  * (b * p.rho[i] * p.alpha_jutzi[i] * p.e[i] / (p.e[i] / (E_0 * eta * eta) + 1.0)
215  + A * mu * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0)))
216  + b * p.e[i] * (1.0 + 3.0 * p.e[i] / (E_0 * eta * eta)) / (pow(p.e[i] / (E_0 * eta * eta) + 1.0, 2))
217  + A * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0))
218  * (1.0 / rho_0 + beta_till / (p.rho[i] * p.alpha_jutzi[i])
219  - beta_till * rho_0 / (p.rho[i] * p.rho[i] * p.alpha_jutzi[i] * p.alpha_jutzi[i])));
220  } else if (p.e[i] > E_iv && eta < 1.0) {
221  /* for intermediate states:
222  * weighted average of pressures calculated by expanded
223  * and compressed versions of Tillotson (both evaluated at e)
224  */
225  p1 = (a + b / (p.e[i] / (eta * eta * E_0) + 1.0))
226  * p.rho[i] * p.alpha_jutzi[i] * p.e[i] + A * mu + B * mu * mu;
227  p2 = a * p.rho[i] * p.alpha_jutzi[i] * p.e[i]
228  + (b * p.rho[i] * p.alpha_jutzi[i] * p.e[i] / (p.e[i] / (eta * eta * E_0) + 1.0)
229  + A * mu * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0)))
230  * exp(-alpha_till * (pow(rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0, 2)));
231  pressure_solid = ((p.e[i] - E_iv) * p2 + (E_cv - p.e[i]) * p1) / (E_cv - E_iv);
232  p.delpdele[i] = ((p2 - p1) + (p.e[i] - E_iv) * a * p.rho[i] * p.alpha_jutzi[i]
233  + p.rho[i] * p.alpha_jutzi[i] * b / (pow(p.e[i]/(E_0 * eta * eta) + 1.0, 2))
234  * exp(-alpha_till * (pow(rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0, 2)))
235  + (E_cv - p.e[i]) * a * p.rho[i] * p.alpha_jutzi[i] + p.rho[i] * p.alpha_jutzi[i]
236  * b / (pow(p.e[i] / (E_0 * eta * eta) + 1.0, 2))) / (E_cv - E_iv);
237  p.delpdelrho[i] = ((a * p.e[i] + exp(-alpha_till * (pow(rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0, 2)))
238  * (2.0 * alpha_till * rho_0 / (p.rho[i] * p.rho[i] * p.alpha_jutzi[i]
239  * p.alpha_jutzi[i]) * (rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0)
240  * (b * p.rho[i] * p.alpha_jutzi[i] * p.e[i] / (p.e[i] / (E_0 * eta * eta) + 1.0)
241  + A * mu * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0)))
242  + b * p.e[i] * (1.0 + 3.0 * p.e[i] / (E_0 * eta * eta)) / (pow(p.e[i] / (E_0 * eta * eta) + 1.0, 2))
243  + A * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_jutzi[i]) - 1.0))
244  * (1.0 / rho_0 + beta_till / (p.rho[i] * p.alpha_jutzi[i]) - beta_till * rho_0
245  / (p.rho[i] * p.rho[i] * p.alpha_jutzi[i] * p.alpha_jutzi[i])))) * (p.e[i] - E_iv)
246  + (a * p.e[i] + p.e[i] * b * (1.0 + 3.0 * p.e[i] / (E_0 * eta * eta))
247  / (pow(p.e[i] / (E_0 * eta * eta) + 1.0, 2))
248  + A / rho_0 + 2.0 * B / rho_0 * (eta - 1.0))
249  * (E_cv - p.e[i])) / (E_cv - E_iv);
250  } else {
251  printf("Deep trouble in pressure.\n");
252  printf("p[%d].e = %e\n", i, p.e[i]);
253  printf("E_iv: %e, E_cv: %e\n", E_iv, E_cv);
254  pressure_solid = 0.0;
255  }
256  }
257  } else if (matEOS[matId] == EOS_TYPE_JUTZI_MURNAGHAN) {
258  double rho_0 = matRho0[matId];
259  double n = matN[matId];
260  double K_0 = matBulkmodulus[matId];
261  double eta = p.rho[i] * p.alpha_jutzi[i] / rho_0;
262  pressure_solid = K_0 / n * (pow(eta, n) - 1.0);
263  p.delpdele[i] = 0.0;
264  p.delpdelrho[i] = K_0 / rho_0 * (pow(eta, n - 1.0));
265  }
266  pressure = pressure_solid / p.alpha_jutzi[i]; /* from the P-alpha model */
267  /* calculate the derivative dalpha / dpressure */
268  double dalphadp_elastic = 0.0;
269  // double c_0 = 5350.0; /* If dalpha_dp_elastic is NOT Zero then you need values for c_0 and c_e */
270  // double c_e = 4110.0;
271  // double h = 1 + (p.alpha_jutzi[i] - 1.0) * (c_e - c_0) / (c_0 * (alpha_e - 1.0)); /* needs to have c_e and c_0 set */
272  // dalphadp_elastic = p.alpha_jutzi[i] * p.alpha_jutzi[i] / (c_0 * c_0 * rho_0) * (1.0 - (1.0 / (h * h)));
273  p.dalphadp[i] = 0.0;
274  if (crushcurve_style == 0) {
275  if (pressure <= p_e) {
276  p.dalphadp[i] = dalphadp_elastic;
277  } else if (pressure > p_e && pressure < p_s) {
278  p.dalphadp[i] = - 2.0 * (alpha_0 - 1.0) * (p_s - pressure) / (pow((p_s - p_e), 2));
279  } else if (pressure >= p_s) {
280  p.dalphadp[i] = 0.0;
281 // p.alpha_jutzi[i] = 1.0;
282  }
283  } else if (crushcurve_style == 1) {
284  if (pressure <= p_e) {
285  p.dalphadp[i] = dalphadp_elastic;
286  } else if (pressure > p_e && pressure < p_t) {
287  p.dalphadp[i] = - ((alpha_0 - 1.0) / (alpha_e - 1.0)) * (alpha_e - alpha_t) * n1 * (pow(p_t - pressure, n1 - 1.0) / pow(p_t - p_e, n1))
288  - ((alpha_0 - 1.0) / (alpha_e - 1.0)) * (alpha_t - 1.0) * n2 * (pow(p_s - pressure, n2 - 1.0) / pow(p_s - p_e, n2));
289  } else if (pressure >= p_t && pressure < p_s) {
290  p.dalphadp[i] = - ((alpha_0 - 1.0) / (alpha_e - 1.0)) * (alpha_t - 1.0) * n2 * (pow(p_s - pressure, n2 - 1.0) / pow(p_s - p_e, n2));
291  } else if (pressure >= p_s) {
292  p.dalphadp[i] = 0.0;
293 // p.alpha_jutzi[i] = 1.0;
294  }
295  }
296  p.dalphadrho[i] = ((pressure / (p.rho[i] * p.rho[i]) * p.delpdele[i] + p.alpha_jutzi[i] * p.delpdelrho[i]) * p.dalphadp[i])
297  / (p.alpha_jutzi[i] + p.dalphadp[i] * (pressure - p.rho[i] * p.delpdelrho[i]));
298  p.f[i] = 1.0 + p.dalphadrho[i] * p.rho[i] / p.alpha_jutzi[i];
299  if (p.alpha_jutzi[i] <= 1.0) {
300  p.f[i] = 1.0;
301  p.alpha_jutzi[i] = 1.0;
302  p.dalphadp[i] = 0.0;
303  p.dalphadrho[i] = 0.0;
304  }
305 #endif
306 #if EPSALPHA_POROSITY
307  } else if (EOS_TYPE_EPSILON == matEOS[matId]) {
308  double pressure_solid = 0.0;
309  double rho_0 = matTillRho0[matId]; /* parameters for Tillotson EOS -> calc pressure solid */
310  double eta = p.rho[i] * p.alpha_epspor[i] / rho_0;
311  double alpha_till = matTillAlpha[matId];
312  double beta_till = matTillBeta[matId];
313  double a = matTilla[matId];
314  double b = matTillb[matId];
315  double A = matTillA[matId];
316  double B = matTillB[matId];
317  double E_0 = matTillE0[matId];
318  double E_iv = matTillEiv[matId];
319  double E_cv = matTillEcv[matId];
320  if (eta < matRhoLimit[matId] && p.e[i] < E_cv) {
321  pressure_solid = 0.0;
322  } else {
323  mu = eta - 1.0;
324  if (p.e[i] <= E_iv || eta >= 1.0) {
325  pressure_solid = (a + b / (p.e[i] / (eta * eta * E_0) + 1.0))
326  * p.rho[i] * p.alpha_epspor[i] * p.e[i] + A * mu + B * mu * mu;
327  } else if (p.e[i] >= E_cv && eta >= 0.0) {
328  pressure_solid = a * p.rho[i] * p.alpha_epspor[i] * p.e[i]
329  + (b * p.rho[i] * p.alpha_epspor[i] * p.e[i] / (p.e[i] / (eta * eta * E_0) + 1.0)
330  + A * mu * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_epspor[i]) - 1.0)))
331  * exp(-alpha_till * (pow(rho_0 / (p.rho[i]
332  * p.alpha_epspor[i]) - 1.0, 2)));
333  } else if (p.e[i] > E_iv && p.e[i] < E_cv) {
334  /* for intermediate states:
335  * weighted average of pressures calculated by expanded
336  * and compressed versions of Tillotson (both evaluated at e)
337  */
338  p1 = (a + b / (p.e[i] / (eta * eta * E_0) + 1.0))
339  * p.rho[i] * p.alpha_epspor[i] * p.e[i] + A * mu + B * mu * mu;
340  p2 = a * p.rho[i] * p.alpha_epspor[i] * p.e[i]
341  + (b * p.rho[i] * p.alpha_epspor[i] * p.e[i] / (p.e[i] / (eta * eta * E_0) + 1.0)
342  + A * mu * exp(-beta_till * (rho_0 / (p.rho[i] * p.alpha_epspor[i]) - 1.0)))
343  * exp(-alpha_till * (pow(rho_0 / (p.rho[i] * p.alpha_epspor[i]) - 1.0, 2)));
344  pressure_solid = ((p.e[i] - E_iv) * p2 + (E_cv - p.e[i]) * p1) / (E_cv - E_iv);
345  } else {
346  printf("Deep trouble in pressure.\n");
347  printf("p[%d].e = %e\n", i, p.e[i]);
348  printf("E_iv: %e, E_cv: %e\n", E_iv, E_cv);
349  pressure_solid = 0.0;
350  }
351  }
352  pressure = pressure_solid / p.alpha_epspor[i]; /* from the P-alpha model which is also used here */
353  p.p[i] = pressure;
354  // printf("Particle: %d \t P: %e \t Alpha: %e \t Rho: %e \t E: %e \t Mu: %e\n", i, pressure, p.alpha_epspor[i], p.rho[i], p.e[i], mu);
355 #endif
356  } else if (EOS_TYPE_REGOLITH == matEOS[matId]) {
357 #if SOLID
358  p.p[i] = 0.0;
359  double I1;
360 #if DIM == 2
361  double shear = matShearmodulus[matId];
362  double bulk = matBulkmodulus[matId];
363  double poissons_ratio = (3.0*bulk - 2.0*shear) / (2.0*(3.0*bulk + shear));
364  I1 = (1 + poissons_ratio) * (p.S[stressIndex(i, 0, 0)] + p.S[stressIndex(i, 1, 1)]);
365 #else
366  I1 = p.S[stressIndex(i,0,0)] + p.S[stressIndex(i,1,1)] + p.S[stressIndex(i,2,2)];
367 #endif
368  p.p[i] = -I1/3.0;
369 #endif
370  } else {
371  printf("No such EOS. %d\n", matEOS[matId]);
372  }
373 #if PALPHA_POROSITY
374  if (matEOS[matId] == EOS_TYPE_JUTZI || matEOS[matId] == EOS_TYPE_JUTZI_MURNAGHAN) {
375  p.p[i] = pressure;
376  } else {
377  p.alpha_jutzi_old[i] = p.alpha_jutzi[i];
378  }
379 #endif
380 #if REAL_HYDRO
381  if (p.p[i] < 0.0)
382  p.p[i] = 0.0;
383 #endif
384  }
